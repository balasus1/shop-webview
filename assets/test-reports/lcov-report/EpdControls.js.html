  initSwipeScrolling() {
 
    const body = document.body;
    const swiped = (e) => {      
      if (e.timeStamp - this.startSwipe.timeStamp < 3000) {
        const evt = e;
        const deltaY = evt.changedTouches[0].pageY - this.startSwipe.y;
        const deltaX = evt.changedTouches[0].pageX - this.startSwipe.x;
        // Neglect swpies which are under 45 degree angle and 100 length
        try {
          if (Math.abs(deltaY) >= 10 && Math.abs(deltaY/deltaX) > 1) {
            return deltaY / Math.abs(deltaY);
          }
        } catch (e) {
          // do nothing
        }
      }
      return 0;
    }
 
    body.ontouchstart = (e) => {
      e = e.timeStamp ? e : window.event; // EPD sucks
      this.startSwipe = {
        timeStamp: e.timeStamp,
        y: e.changedTouches[0].pageY,
        x: e.changedTouches[0].pageX
      }
    };
 
    body.ontouchcancel = (e) => {
      e = e.timeStamp ? e : window.event; // EPD sucks
      const direction = swiped(e);
      switch (direction) {
        case -1: this.pageDown(); break;
        case 1: this.pageUp(); break;
        default: return;
      }
    };
 
    body.ontouchend = (e) => {
      e = e.timeStamp ? e : window.event; // EPD sucks
      const direction = swiped(e);
      switch (direction) {
        case -1: this.pageDown(); break;
        case 1: this.pageUp(); break;
        default: return;
      }
    };
  }

  pageDown(e) {

    let scrollHeight = this.getScrollHeight();

    //Math.ceil is used to round off to 1 even if totalPages has values between 0.10 to 0.99.

  }
