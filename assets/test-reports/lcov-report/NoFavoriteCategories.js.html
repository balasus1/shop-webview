import React from 'react'
 
import platforms from '../lib/constants/platforms'
 
import { constants, getImageName } from '../lib/utils'
import defaultBook from "../../../public/images/categories_icons_medium/default.svg";
 
class NoFavoriteCategories extends React.Component {
    constructor(props) {
        super(props);
        this.isEPD = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
        this.state = {
            isTab: this.isEPD || window.innerWidth > 480
        };
    }
 
    componentDidMount() {
        this.isTabView();
        window.addEventListener("resize", this.isTabView);
    }
 
    componentWillUnmount() {
        window.removeEventListener("resize", this.isTabView);
    }
 
    isTabView = () => {
        this.setState({ isTab: this.isEPD || window.innerWidth > 480 });
    }
 
    render() {
        const isEPD = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
        let noFavCategoriesIcons = ["29", "6727", "53", "19217", "6730"];
        if (this.state.isTab) {
            noFavCategoriesIcons = ["29", "6727", "214", "53", "1760"];
            noFavCategoriesIcons.push("19217","6730");
        }
        let categoryImage = null;
        let categorySticker = [];
 
 
        for (let index = 0; index < noFavCategoriesIcons.length; index++) {
            let cat = noFavCategoriesIcons[index];
            const getFileName = (arg) => {
                try {
                    const images = require.context('../../../public/images/categories_icons_medium', true);
                    let imgsrc = images(`./${arg}`);
                    categoryImage = imgsrc;
                }
                catch (error) {
                    categoryImage = defaultBook;
                }
                categorySticker[index] = <img key={index} src={categoryImage} className="category-icon" />;
            }
 
            getImageName({ id: cat }, getFileName);
        }
 
        return (
            <React.Fragment>
                <div className={`static-content ${this.state.isTab ? 'tab-view' : ''}`}>
                    <div className="title">{constants.EXPLORE_FAVORITES_TITLE}</div>
                    <div className="explore-fav-usage">{constants.EXPLORE_FAVORITES_USAGE_INFO}</div>
                    <div className="lets-go-button-container">
                        <button className={"edit-fav first-time-user " + (isEPD ? "epd" : "")} onClick={this.props.editFavCategories}>
                            {constants.SETUP_FAVORITES_BUTTON_TEXT}
                        </button>
                    </div>
                    <div className="no-fav-img-container">
                        {categorySticker}
                    </div>
                </div>
            </React.Fragment>
        );
    }
}
 
export default NoFavoriteCategories;
