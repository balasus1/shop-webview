import { connect } from 'react-redux'
import actionTypes from '../lib/constants/actionTypes'
import ForMe from '../components/ForMe'
 
const mapStateToProps = (state, ownProps) => {
  return {
    customer: state.sessionData.customer,
    device: state.sessionData.device,
    eanData: state.eanData,
    data: state.forMeData,
    affinities: state.affinityData.affinities.slice(0, 5) // We only care about the top 5
  }
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
 
  // This function updates our global EAN metadata store in redux
  const updateEanData = (eanData) => {
    dispatch({
      type: actionTypes.ADD_EAN_DATA,
      eans: eanData
    });
  }
 
  return {
 
    // Update recommendations carousel redux state
    setF1: (data) => {
      updateEanData(data.eanData);
      dispatch({
        type: actionTypes.SET_F1_DATA,
        f1Data: data.data
      });
    },
 
    // Update pods redux state
    setF2: (data) => {
      updateEanData(data.eanData);
      dispatch({
        type: actionTypes.SET_F2_DATA,
        f2Data: data.data,
        status: data.status
      });
    },
 
    setFavCategories: (data) => {
      dispatch({
        type: actionTypes.SET_FAV_CATEGORIES_DATA,
        favCategoriesData: data
      });
    },
 
    updateFavCategories: (data) => {
      dispatch({
        type: actionTypes.UPDATE_FAV_CATEGORIES_DATA,
        favCategoriesData: data
      });
    },
 
    // Update links (and categories) redux state
    setF3: (data) => {
      updateEanData(data.eanData);
      dispatch({
        type: actionTypes.SET_F3_DATA,
        f3Data: data.data,
        status: data.status
      });
    },
 
    // Update lists redux state
    setF4: (data) => {
      updateEanData(data.eanData);
      dispatch({
        type: actionTypes.SET_F4_DATA,
        f4Data: data.data,
        status: data.status
      });
    },
 
    // Update more lists for you redux state
    setF5: (data) => {
      updateEanData(data.eans);
      dispatch({
        type: actionTypes.SET_F5_DATA,
        f5Data: data.affinityLists,
      });
    },
 
 
    // Build a synthetic list from the F1 carousel data since we cannot
    // fetch the EANs from the assembler (it's not a real list).
    buildF1List: (listData) => {
      dispatch({
        type: actionTypes.SET_LIST_DETAILS,
        listId: listData.listId,
        listTotal: null, // Do not render a count for this "list"
        listTitle: listData.listTitle,
        listDescription: listData.listDescription
      });
 
      dispatch({
        type: actionTypes.SET_LIST_ITEMS,
        listData: {
          loadedListId: listData.listId,
          eans: listData.eans
        }
      });
    }
  }
}
 
const ForMeContainer = connect(
  mapStateToProps,
  mapDispatchToProps
)(ForMe)
 
export default ForMeContainer
