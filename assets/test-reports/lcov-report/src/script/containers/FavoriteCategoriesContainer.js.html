import { connect } from 'react-redux'
import actionTypes from '../lib/constants/actionTypes'
import FavoriteCategories from '../components/FavoriteCategories'
 
const mapStateToProps = (state, ownProps) => {
    return {
      customer: state.sessionData.customer,
      device: state.sessionData.device,
      data: state.favCategoriesData,
    }
  }
  
  const mapDispatchToProps = (dispatch, ownProps) => {
    return {
      setFavCategories: (data) => {
        dispatch({
          type: actionTypes.SET_FAV_CATEGORIES_DATA,
          favCategoriesData: data
        });
      },
      updateFavCategories: (data) => {
        dispatch({
          type: actionTypes.UPDATE_FAV_CATEGORIES_DATA,
          favCategoriesData: data
        });
      },
    }
  }
  
  const FavoriteCategoriesContainer = connect(
    mapStateToProps,
    mapDispatchToProps
  )(FavoriteCategories);
 
  export default FavoriteCategoriesContainer;
