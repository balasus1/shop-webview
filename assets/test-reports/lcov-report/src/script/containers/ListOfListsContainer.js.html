import { connect } from 'react-redux'
import actionTypes from '../lib/constants/actionTypes'
import ListOfLists from '../components/ListOfLists'
 
const mapStateToProps = (state, ownProps) => {
  return {
    customer: state.sessionData.customer,
    device: state.sessionData.device,
    eanData: state.eanData,
    data: Object.assign({}, state.lolData)
  }
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
 
  const updateEanData = (eanData) => {
    dispatch({
      type: actionTypes.ADD_EAN_DATA,
      eans: eanData
    });
  };
 
  return {
    
    // Set Nook Channels data in redux
    setLists: (lolData) => {
      updateEanData(lolData.eans);
 
      dispatch({
        type: actionTypes.SET_LOL_ITEMS,
        lists: lolData.lists,
      });
    }
  }
}
 
const ListOfListsContainer = connect(
  mapStateToProps,
  mapDispatchToProps
)(ListOfLists)
 
export default ListOfListsContainer
