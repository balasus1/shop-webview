const KEY_OTA_DOWNLOAD_PERCENTAGE = "OTADownloadPercentage"
class EpdUpgrade  extends Component {
  downloadStatusIntervalId = 0
  downloadTimeoutIntervalId = 0
  showDownloadErrorIntervalId = 0
  onUpdateIntervalId = 0
  constructor(props) {
    super(props);
    //working
    
    this.state = {
      isDownloaded:false,
      isDownloading: false,
      downloadProgress: 0,
      downloadError: false,
      downloadTimeout: false,
      isNetworkConnected: true,
      showDownloadError: false
    } 
    
    this.openUpdate = this.openUpdate.bind(this);
    this.handleDownloadPercent = this.handleDownloadPercent.bind(this);
    this.handleDownloadPercentNew = this.handleDownloadPercentNew.bind(this);
    this.initNativeHandlers = this.initNativeHandlers.bind(this);
    this.updateDownloadStatus = this.updateDownloadStatus.bind(this);
    this.downloadTimeout = this.downloadTimeout.bind(this);
    this.showDownloadError = this.showDownloadError.bind(this);
    this.getDownloadStatus = this.getDownloadStatus.bind(this)
    this.onUpdate = this.onUpdate.bind(this)
    this.listenNetworkChange = this.listenNetworkChange.bind(this)
    this.downloadStatusIntervalId = setInterval(this.getDownloadStatus, 1000);
    this.initAnalytics()
  }
 
 
  initAnalytics() {
    const defaultValues = {
      [attributes.errorCode]: 'com.bn.NookApplication.YourNookWebview #101',
      [attributes.activeScreen]: 'YOUR_NOOK',
      [attributes.activeComponent]: 'HomeWebView',
      [attributes.errorDetails]: process.env.REACT_APP_VERSION,
    };
 
    this.updateAvailableEvent = Object.assign({}, events.errorOccurred);
    this.updateAvailableEvent.data = Object.assign({}, events.errorOccurred.data, defaultValues);
    this.updateAvailableEvent.data[attributes.errorDescription] = '4.9.5 Update Available';
 
    this.updateStartedEvent = Object.assign({}, events.errorOccurred);
    this.updateStartedEvent.data = Object.assign({}, events.errorOccurred.data, defaultValues);
    this.updateStartedEvent.data[attributes.errorDescription] = '4.9.5 Update Started';
 
    this.updateSuccessfulEvent = Object.assign({}, events.errorOccurred);
    this.updateSuccessfulEvent.data = Object.assign({}, events.errorOccurred.data, defaultValues);
    this.updateSuccessfulEvent.data[attributes.errorDescription] = '4.9.5 Update Successful';
 
    this.updateFailedEvent = Object.assign({}, events.errorOccurred);
    this.updateFailedEvent.data = Object.assign({}, events.errorOccurred.data, defaultValues);
    this.updateFailedEvent.data[attributes.errorDescription] = '4.9.5 Update Failed';
  }
 
 
  updateDownloadStatus() {
 
    if (this.state.downloadProgress && this.state.downloadProgress.toString() === '100') {
      const isDownloadComplete = (localStorage.getItem('downloadComplete') === 'true') ? true : false
      this.setState({ isDownloaded: isDownloadComplete })
      return isDownloadComplete;
    }
    
  }
 
  getDownloadStatus() {
    const progressPercentage = localStorage.getItem(KEY_OTA_DOWNLOAD_PERCENTAGE);
    if (progressPercentage === '100') {
      this.setState({isDownloaded:true})
      this.showDownloadErrorIntervalId = window.setInterval(this.showDownloadError, 20000);
    }
    
    if (progressPercentage>0) {
      this.setState({isDownloading:true , downloadProgress : progressPercentage})
    }
  }
 
  downloadTimeout() {
    console.log('downloadTimeout111111111111111111111',this.state.downloadProgress);
    if (this.state.downloadProgress === '0') {
      console.log('downloadTimeout2222222222222222',this.state.downloadTimeout);
      this.setState({downloadTimeout: true})
    }
  }
 
 
  componentWillMount () {
    this.initNativeHandlers();
    console.log('localstorageDownloadStatus',sessionStorage.getItem('showDownloadError'))
  }
 
  componentWillUnmount() {
    console.log('------------------componentWillUnmount----------------------',this.state.isDownloaded)
    clearInterval(this.downloadStatusIntervalId)
    clearInterval(this.downloadTimeoutIntervalId)
    clearInterval(this.showDownloadErrorIntervalId)
    clearInterval(this.onUpdateIntervalId)
  }
 
  componentDidMount() {
    this.listenNetworkChange();
    this.sendAnalytics(this.updateAvailableEvent);
  }
 
  componentDidUpdate() {
    document.addEventListener('downloadPercentage', this.handleDownloadPercentNew);
  }
 
  initNativeHandlers() {
    this.setState({ downloadProgress: 0 });
    this.setState({ isDownloading: false });
    localStorage.setItem(KEY_OTA_DOWNLOAD_PERCENTAGE,0);
    window.addEventListener('downloadPercentage', this.handleDownloadPercentNew);
  }
 
  handleDownloadPercentNew(e) {
    //this.setState({ downloadProgress: e.detail.percent });
    console.log('**********************handleDownloadPercentNew++++++++++++',e);
  }
 
  handleDownloadPercent() {
    const isDownloadComplete = localStorage.getItem('downloadComplete')
    this.setState({ isDownloaded: isDownloadComplete });
    const progressPercentage = localStorage.getItem(KEY_OTA_DOWNLOAD_PERCENTAGE);
    this.setState({ downloadProgress: progressPercentage });
    if (progressPercentage === 100) {
      this.setState({ isDownloaded: true });
    }
    if (progressPercentage >> 0) {
      this.setState({ isDownloading: true });
    }
  }
 
  showDownloadError() {
    if (!this.state.showDownloadError) {
      this.setState({showDownloadError:true});
      sessionStorage.setItem('showDownloadError', true);
      console.log('showDownloadError',this.state.showDownloadError);
      clearInterval(this.showDownloadError)
      this.sendAnalytics(this.updateFailedEvent);
    }
  }
 
  onUpdate() {
    this.setState({ downloadProgress: (this.state.downloadProgress < 100) ? localStorage.getItem(KEY_OTA_DOWNLOAD_PERCENTAGE) : 100 })
      window.Device.isConnected(connected => {
        if (connected !== this.state.isNetworkConnected)
          this.setState({ isNetworkConnected: connected })
      });
 
      if (this.state.downloadProgress === 100) {
        this.showDownloadErrorIntervalId = setInterval(this.showDownloadError, 20000);
        clearInterval(this.onUpdate)
      }
  }
 
  listenNetworkChange() {
    if (this.state.downloadProgress < 100) {
      this.onUpdateIntervalId = setInterval(this.onUpdate, 1000)
    }
  }
 
  openUpdate() { 
    window.JSNI.downloadItem('{"ean":"9992222222999", "productType":4}');
    this.setState({ isDownloading: true });
    this.downloadTimeoutIntervalId = setInterval(this.downloadTimeout, 55000);
    this.sendAnalytics(this.updateStartedEvent);
  }
 
  render() {
    let updateIcon = upgradeApp;
    const {isDownloading, isDownloaded, downloadProgress, isNetworkConnected} = this.state;
    const downloaderStyle = {};
    const updateButton = {};
    const downloded = {};
    //console.log('this.state.downloadProgress',this.state.downloadProgress);
 
    if (isDownloading === true) {
      downloaderStyle.width = (downloadProgress * 2) + 'px';
      if (this.state.showDownloadError === true) {
        updateButton.display = 'none'
      }
    }
 
    else if (!isDownloading) {
      downloaderStyle.display = 'none';
    }
 
    if (this.state.showDownloadError === false) {
      downloded.display = 'none'
    }
 
    if (this.state.downloadTimeout===true) {
      updateButton.display = 'none'
    }
    
    return(
      <center>
          <div> 
            <div style={ updateButton }>
              <img src={ updateIcon } className="updateImg" alt="Unable to find item" />
              <div className="updateTitle">A new version is available</div>
              <div className="subTitle"> Performance and stability improvements </div>
              <center className="progresbar">
                <div className={"progressBorder" + (isDownloading === true ? " downloaded" : " notDownloaded")}>
                  <div className={ 'download-button' + 
                    (downloadProgress > 0 && !isDownloaded ? ' downloading' : '') + 
                    (isDownloading === true ? ' downloaded' : 'notDownloaded') }
                    
                    onTouchStart={ e => e.stopPropagation() } style={ downloaderStyle } > 
                  </div>
                </div>
                <div className={ isDownloading === true ? "hideUpdate" : "update"}  onClick={this.openUpdate}> Update </div>
              </center>
              <div className={(isNetworkConnected === false ? 'showMessage' : 'hideMessage')}> 
                <img src={ errorInfo } className="errorInfoImg" alt="Unable to find item" />
                <div className="errorInfo"> <strong>Error</strong> : Unable to access the network </div>
              </div>
              <div className={(isDownloading === true && isNetworkConnected === true ? 'showMessage' : 'hideMessage')}>
                Please press <strong>install</strong> to continue after downloading 
              </div>
              
            </div>
            {(this.state.showDownloadError === true ) ? <div style={ downloded }> 
              <div className="updateTitle"> Update Failed :( </div>
              <div className="subTitle"> An error occurred while installing the update. </div>
              <div className="subTitle"> Please Reboot the device. </div>
              <div className="subTitle"> If the problem persists, please contact customer service. </div>
            </div> : ''}
            {(this.state.downloadTimeout===true) ? <div> 
              <div className="updateTitle"> Update Failed :( </div>
              <div className="subTitle"> An error occurred while installing the update. </div>
              <div className="subTitle"> Please Reboot the device. </div>
              <div className="subTitle"> If the problem persists, please contact customer service. </div>
            </div> : ''}
            
          </div>
      </center>
    )
  }
