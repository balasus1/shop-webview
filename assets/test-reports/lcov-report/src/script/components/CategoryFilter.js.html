class CategoryFilter extends Component {
 
  constructor(props) {
    super(props);
 
    this.state = {
      isFetching: false
    };
    this.isEPD = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
    // Bind methods
    this.renderBrowseLink = this.renderBrowseLink.bind(this);
    this.closeNativeModal = this.closeNativeModal.bind(this);
    window.addEventListener('onScrollToPage', this.setSubCategoriesHeight);
  }
 
  setSubCategoriesHeight(e) {
    let topMargin = window.location.href.match(platforms.epd) ? 37.86 : 51.5;
    if (window.subCategories && window.location.href.match("filter")) {
      const wrapper = document.getElementById('shop-wrapper');
      let landingHeight = wrapper.clientHeight - topMargin;
      window.subCategories.style.height = (landingHeight * e.detail.page) + "px";
      wrapper.scrollTop = (e.detail.page - 1) * (landingHeight - 50);
    }
  }
  
  
  componentWillMount() {
 
    // Show the native back button, set the page title
     const title = this.props.data.drefHash[this.props.match.params.catDref] ?
      this.props.data.drefHash[this.props.match.params.catDref].title : '';
    if (title) {
      window.Device.modifyHeader({
        enableBack: true,
        title: title,
        background: 'E2E7F2',
        hideTools: true
      });
    }
    if(this.isEPD && window.paginationComponent) {
      window.paginationComponent.gotoPage(1); // Start Sub Categories from Page 1.
    }
  }
  
  
  componentDidMount () {
 
    // Analytics
    // (This screen does not have its own event)
    analytics.enter(window.SHOW_AUDIOBOOKS ? screens.audioBooks.category_filter : screens.ebooks.category_filter);
  }
 
 
  isPhone() {
    return window.Device.getMinDimension() < 500;
  }
 
 
  renderBrowseLink(cat, i) {
    if(!cat) { 
      console.log("Category Filter: renderBrowseLink No cat ");
      return;
    }
    return <div key={i} className="category-link"
        onClick={ evt => {
          evt.stopPropagation();
          this.showList(cat.id);
        }}>
      <div> 
        <div className="name">{ cat.label + (window.NOOK_DEBUG ? ' ('+cat.id+')' : '') }</div>
        <img className="categories-icon-right" src={rightArrow} alt={textReplace(KEY.right_arrow)} />
      </div>
    </div>;
  }
 
 
  render() {
    const params = this.props.match.params;
 
    let currentCategory = this.props.data.drefHash[params.catDref];
    let subCategories = this.props.subCategories;
    
    if (!currentCategory) {
      currentCategory = JSON.parse(sessionStorage.getItem("currentcategory"));
    }
 
    return <div className="category-filter">
      <div className='header'>
        <div className="title ellipses">
          {textReplace(KEY.explore) + " " + (currentCategory ? currentCategory.label : "") + (window.NOOK_DEBUG ? ' (' + params.catDref + ')' : '')}
        </div>
          <div className='image-container' onClick={this.closeNativeModal}>
            <img src={closeButton} alt={textReplace(KEY.close_sub_categories)}></img>
          </div>
 
      </div>
 
      <div className="filter-links"  ref={el => window.subCategories = el}>
        { subCategories.length ? subCategories.map(this.renderBrowseLink) :
          <center><Spinner /></center>
        }
        <div className="category-link see-all" ref={el => window.seeAll = el}>
          <div>
            <img className="all-categories-icon" src={AllCategories} alt={textReplace(KEY.see_all_categories)} />
            <div className="name" onClick={() => this.showList('All')}>{textReplace(KEY.see_all_categories).toUpperCase()}</div>
          </div>
        </div>
      </div>
    </div>;
  }
 
 
  showList(dref) {
    window.categoryPageReset = false;
    let hash = window.location.hash.split('/');
    hash.pop();
    hash.pop();
    hash.pop();
 
    if (dref === 'All') {//Navigate to All categories Menu
      const params = this.props.match.params;
      let store = window.location.hash.split(params.catStoreCode)[0];
      hash = store + (params.catStoreCode === "kids" ? (params.catStoreCode + "/" + params.catDref.split(",")[0]) : params.catStoreCode);
    } else {
      hash = `${hash.join('/')}/${dref}/list`
    }
 
    if (window.LOCALHOST) {//development
      window.location.hash = hash;
    }
    else {//call native
      window.Device.closeModalGotoShop({ categoriesRelativePath: hash });
    }
  }

    if(window.LOCALHOST) {
      window.history.back()
    } else {
      window.Device.closeSubCategoriesModal();
    }
