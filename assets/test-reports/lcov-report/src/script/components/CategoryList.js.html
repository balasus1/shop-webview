    this.state = {

  componentWillMount() {
    this.loadCategories();
    this.title = this.props.data.drefHash[this.props.match.params.catDref] ?
      this.props.data.drefHash[this.props.match.params.catDref].label : '';
    if (!this.isEPD && this.title == constants.KIDS_BOOKS) {
      this.title = window.SHOW_AUDIOBOOKS ? constants.KIDS_AUDIOBOOKS :
        constants.KIDS_EBOOKS
    }
    window.location.hash = window.location.hash.replace('/filter', '/filter');
    // Show the native back button, set the page title
    window.Device.modifyHeader({
      enableBack: true,
      title: this.title,
      hideTools: true
    });
    // Fetch the spots (merch lists) and refinements (child DREFs) for the current DREF
    this.loadCategoryData(this.props.match.params.catStoreCode, this.props.match.params.catDref);
    sessionStorage.setItem('categoriesPosition', 0);
    var position = sessionStorage.getItem("categoriesPosition") || 0;
    window.scroll(0, position);
    const catDref = this.props.match.params.catDref;
    if (this.props.match.params.filterSort === 'filter') {
      this.setState({ showFilter: true });
    }
 
    if (this.props.data.drefHash[catDref]) {
      sessionStorage.setItem("currentcategory", JSON.stringify(this.props.data.drefHash[catDref]));
    }
    if(this.isEPD && window.paginationComponent) {
      window.paginationComponent.gotoPage(sessionStorage.getItem("epd_category_list_page") || 1);
    }
  }
 
  loadCategories() {
    let currentStore = storeCodes[storeCodeMap[this.props.customer.childProfile ? "kids" : "nkbk"]] || null
   
    this.setState({ isFetchingCategories: true });
 
    // Callback function - fetch complete
    const done = (err, response) => {
      if (err) {
        console.error('Error fetching category data.', err.message);
        window.Device.showError('Error fetching category data. ' + err);
        response = { eans: [], categories: [], links: [] };
      }
      else {
        this.setState({ isFetchingCategories: false });
      }
      const catData = {
        data: getCurrentCategories(response, this.props.customer.childProfile, window.SHOW_AUDIOBOOKS),
      };
      this.props.setStoreCategories(catData, currentStore.code); // Trigger a redux dispatch
    };
    fetchStoreCategories(currentStore.code, {}, done); // Cached for 12 hrs
  }
 
  componentDidMount() {
    // Hack to trigger scrolling at the right position....
    sessionStorage.setItem('categoriesPosition', 0);
    sessionStorage.getItem("categoriesPosition");
    // Analytics
    if (!this.state.showFilter) {
      analytics.enter(window.SHOW_AUDIOBOOKS ? screens.audioBooks.category_results : screens.ebooks.category_results);
      this.initAnalytics();

    if (!this.isEPD) {
      this.sticky();

    if(this.isEPD) {
      this.generateFillers();
    }

  sticky = () => {
    //Intersection Observer
    const handler = (entries) => {
      this.setState({ intersecting: entries[0].isIntersecting });
    }
    const observer = new window.IntersectionObserver(handler)
 
    if (this.sentinal)
      observer.observe(this.sentinal)
  }
 
  componentWillUpdate(nextProps, nextState) {
    // Show/hide the filter menu
    if (nextProps.match.params.filterSort !== this.props.match.params.filterSort) {
      this.setState({
        showFilter: nextProps.match.params.filterSort === 'filter'
      });
    }
  }
 
 
  componentDidUpdate(prevProps, prevState) {
    if (prevProps.match.params.catDref !== this.props.match.params.catDref) {
      if (!this.isEPD) {
        this.resetToTop();
      }
      else if(window.paginationComponent) {
        window.paginationComponent.gotoPage(1); // Start Sub Categories from Page 1.
      }
      this.loadCategoryData(this.props.match.params.catStoreCode, this.props.match.params.catDref);
    }
    this.handleCategoriesList();
  }
 
 
  componentWillUnmount() {

    if(window.paginationComponent) {
      sessionStorage.setItem("epd_category_list_page", window.paginationComponent.getPageNo())
    }
  }
 
 
  initAnalytics() {
    this.analyticsEvent = Object.assign({}, events.categoryViewed);
    this.analyticsEvent.data = Object.assign({}, events.categoryViewed.data);
  }
 
  resetToTop() {
    window.scroll(0,0); //start from top of the page.
    let interval = setInterval(() => {
      if (document.readyState === 'complete') {
        clearInterval(interval);
        window.scroll(0, 0);
      }
    }, 300);
  }
  loadCategoryData(catStoreCode, catDref, options = {}) {
 
    // Don't fetch data we already have
    if (this.props.data.categories[catDref]) {
      return;
    }
 
    let device = this.props.device;
    let customer = this.props.customer;
    let urlParams = {
      countryCode: customer.country,
      accountId: customer.accountId,
      profileId: customer.profileId,
      productDeviceId: device.productDeviceId,
      profileType: customer.childProfile ? "kids" : "adult",
      shopHome: window.SHOW_AUDIOBOOKS ? "audiobook" : "ebook"
    };
 
    // Callback function - fetch complete
    const done = (err, response) => {
      if (err) {
        console.error('Error fetching category data.', err.message);
        window.Device.showError(textReplace(KEY.categorylist_error) + err.message);
        response = { eans: [], categories: [], links: [] };
      }
      else {
        this.setState({ isFetching: false });
      }
 
      const catData = {
        data: response,
        eanData: response.eans
      };
 
      this.props.setSubCategories(catData, catDref); // Trigger a redux dispatch
    };
 
    fetchSubCategories(catStoreCode, catDref, urlParams, done, options); // Cached 12 hrs if option.cacheResponse
  }
 
 
  renderTitle() {
    const params = this.props.match.params;
    const drefHash = this.props.data.drefHash;
    if (!drefHash || !drefHash[params.catDref]) {
      return '';
    }
 
    const store = storeCodes[storeCodeMap[params.catStoreCode]];
    const drefs = params.catDref.split(',');
    this.headerTitle = (drefHash[params.catDref] || {}).label;
    let children = drefHash[params.catDref].children;
    let canFilter = (children || []).length > 0;
    this.state.canFilter = canFilter;
    if (this.headerTitle) {
      window.Device.modifyHeader({
        enableBack: true,
        title: ((drefs && drefs.length === 1) && !this.isEPD) ? window.Device.getStoreNameForSubCategories(store) : this.headerTitle,
        hideTools: true
      });
    }
 
    const catDref = this.props.match.params.catDref;
    const currentCategory = this.props.data.drefHash[catDref];
    const currentCategoryTitle = currentCategory ? currentCategory.label : "";
    const parentDref = (currentCategory && currentCategory.parent) ? currentCategory.parent : -1;
    const parentTitle = this.props.data.drefHash[parentDref] ? this.props.data.drefHash[parentDref].label : "";
   
    const filterContainer =
      <div  ref={elt => this.stickyDiv = elt} className={"filter-container " + (!this.state.intersecting ? ' sticky' : '')} onClick={this.openFilter}>
        <div className="filter-icon"><img src={filterIcon} alt={textReplace(KEY.filter_icon)} /></div>
        <div className="filter-text ellipses lines-1">{textReplace(KEY.explore).toUpperCase() + " " + currentCategoryTitle}</div>
      </div>
 
 
    const categoryHeader =
    (!this.isEPD || canFilter ) ? <div className={"category-header"}
        ref={el => this.categoryHeader = el}>
        {!this.isEPD && parentTitle && <div className='parent-title'>{parentTitle}</div>}
        {!this.isEPD && <div className="category-title ellipses lines-1">
          {currentCategoryTitle}
        </div>}
        {canFilter ? filterContainer : null}
      </div>: "";
 
    return <React.Fragment>
      {categoryHeader}
    </React.Fragment>;
  }
 
  renderCarousel(list, idx, loading, isPerfectBook) {

    const searchParams = this.getSearchParams(list.id);
    if (idx === 3) {
      list.type = constants.TWO_ROWS;

        <ListCarousel key={idx}
          aria-label="0"
          useSearch
          loading={loading}
          search={searchParams}
          index={idx}
          list={list}
          eanData={props.eanData}
          isCategoryPage={window.location.href.match('/categories/')}
          onCoverClick={data => this.prepareAnalytics('cover', data)}
          onListClick={data => this.prepareAnalytics('list', data)} />
      }
      </React.Fragment>);
  }

  generateFillers(data, noOfRows) {

        fillers.push(<ContainerFiller key={i} fillContainer="list-carousel" />);
      }
    }
    return fillers;
  }

        {<ListContainer noModifyHeader noAnalytics useDref countInTitle = {!this.isEPD}

          hasFilter={this.state.canFilter}
          hasExploreCategoriesFilter={ this.state.canFilter }
          defaultSort="sa"
          defaultAudFilter="dig"//audiobook filter
          showSort={showSort}
          storeCode={params.catStoreCode}
          dref={params.catDref}
          onDemandLoad={demandLoadCount => this._demandLoadCount = demandLoadCount}
          onCoverClicked={resultsEanPosition => this.prepareAnalytics('all_results', { resultsEanPosition })}
          showAllCategories={this.showAllCategories} />}
        {isCategories && !this.isEPD && <div className="show-all-categories-container" ref={el => window.seeAll = el}>
          <img className="all-categories-icon" src={AllCategories} alt={textReplace(KEY.see_all_categories)} />
          <div className="name" onClick={this.showAllCategories}>{textReplace(KEY.see_all_categories)}</div>
        </div>}
      </div> :
      <div className={'filter-sidebar' + (this.state.showFilter ? ' active' : '')}>
        {<CategoryFilter
          showAllCategories={this.showAllCategories}
          match={this.props.match}
          data={this.props.data}
          subCategories={subCategories} />
        }
      </div>;
  }
 
 
  getSearchParams(searchURL) {
    let searchParams = searchURL.split('?').pop();
 
    // "sort" can't be the first param or it opens the sort menu in the list when opened
    const firstParam = searchParams.split('&')[0];
    if (firstParam.match('sort=')) {
      searchParams = searchParams.replace(firstParam, '') + '&' + firstParam;
    }
    return searchParams;
  }
 
  openFilter() {
    // Set analytics data
    this.prepareAnalytics('filter');
    window.categoryPageReset = false;
    
    if (window.LOCALHOST && !window.location.hash.match('/filter')) {//development purpose
      window.location.hash += '/filter';
    }
    else {//native
      console.log(this.props.match.url + '/filter')
      window.Device.openSubcategoriesModal({categoriesRelativePath: "#" + this.props.match.url + '/filter'})
    }
  }
 
  prepareAnalytics(type, data) {
    if (this.analyticsEvent) {
      this.analyticsEvent.data[attributes.shopType] = window.Device.getShopType();
 
      switch (type) {
 
        case 'filter':
          this.analyticsEvent.data[attributes.resultsSection] = 'filter';
          break;
 
        case 'cover':
          this.analyticsEvent.data[attributes.resultsSection] = 'carousel';
          this.analyticsEvent.data[attributes.listEanPosition] = data.listEanPosition;
          this.analyticsEvent.data[attributes.listTitle] = data.listTitle;
          this.analyticsEvent.data[attributes.listViewAllClicked] = 'No';
          this.sendAnalytics(); // The componented doesn't unmount; call manually
          break;
 
        case 'list':
          this.analyticsEvent.data[attributes.resultsSection] = 'carousel';
          this.analyticsEvent.data[attributes.listTitle] = data.listTitle;
          this.analyticsEvent.data[attributes.listViewAllClicked] = 'Yes';
          break;
 
        case 'all_results':
          this.analyticsEvent.data[attributes.resultsSection] = 'all results';
          this.analyticsEvent.data[attributes.resultsEanPosition] = data.resultsEanPosition;
          this.sendAnalytics(); // The componented doesn't unmount; call manually
          break;
        default:
          console.error('Invalid analytics type. What did you do wrong??');
      }
    }
  }
 
  sendAnalytics() {
 
    if(this.analyticsEvent && this.analyticsEvent.data) {
      // Set values
      this.analyticsEvent.data[attributes.categoryName] = this.title;
      this.analyticsEvent.data[attributes.demandLoadCount] = this._demandLoadCount;
    }
 
    // Send the event
    analytics.exit(this.analyticsEvent);
 
    // Reset the event data since we'll come back from PDP
    this.initAnalytics();
  }
 
}
