  loadContent(urlParams, options) {
    const props = this.props;
 
    // Load Pods
    this.setState({ isFetchingF2: true });
    fetchF2(urlParams, (err, resp) => {
      if (err) {
        console.error('Error fetching F2 pods.');
        resp = {eans:[], pods:[]};
      }
      this.setState({ isFetchingF2: false });
 
      const data = {
        data: resp.pods,
        eanData: resp.eans,
        status: resp.status
      };
      props.setF2(data); // Trigger a redux dispatch
    }, options);
 
    // Load Explore favorites
    this.setState({ isFetchingFavCategories: true });
      urlParams.shopHome = this.shopHome;
      favoriteCategories('get', urlParams, null, (err, resp) => {
        if (err) {
          console.error('Error fetching Favorite Categories.');
          resp = {favorites: []};
        }
        this.setState({isFetchingFavCategories: false});
        if(resp && resp.data) {
          let responseData = this.shopHome === "ebook" ? resp.data.ebook : resp.data.audiobook;
          if(!responseData) { // Remove this check once NOOKCLOUD-5665 goes to prod
            responseData = resp.data.favorites;
          }
 
          if (responseData) {
            props.setFavCategories(responseData); // Trigger a redux dispatch
          } else if(localStorage.getItem(this.selectedFavCategoriesKey)) {
            props.setFavCategories(JSON.parse(localStorage.getItem(this.selectedFavCategoriesKey)));

  prepareAnalytics(type, data) {
    sessionStorage.setItem('forMePosition', window.scrollY);
 
    this.analyticsEvent.data[attributes.shopType] = window.Device.getShopType();;
 
    if(this.isEpd) {
      sessionStorage.setItem('forMePageNo', window.paginationComponent.getPageNo());
    }
    switch (type) {
      case 'pod':
        sessionStorage.setItem('f2PodPosition', data.f2Position-1);
        this.analyticsEvent.data[attributes.f2Title] = data.f2Title;
        this.analyticsEvent.data[attributes.f2Position] = data.f2Position;
        this.analyticsEvent.data[attributes.shopSection] = 'F2';
 
        // Single EAN pod opens PDP
        if (data.promotionEan) {
          this.analyticsEvent.data[attributes.promotionEan] = data.promotionEan;
          this.sendAnalytics(); // The componented doesn't unmount; call manually
        }
        break;
 
      case 'favoriteCategories':
        sessionStorage.setItem('favCatPosition', data.favCatPosition-1);
        this.analyticsEvent.data[attributes.favCatTitle] = data.favCatTitle;
        this.analyticsEvent.data[attributes.favCatPosition] = data.favCatPosition;
        this.analyticsEvent.data[attributes.shopSection] = 'Favorite Categories';
        this.sendAnalytics();
        break;
 
      case 'link':
        this.analyticsEvent.data[attributes.f3Title] = data.f3Title;
        this.analyticsEvent.data[attributes.f3Position] = data.f3Position;
        this.analyticsEvent.data[attributes.shopSection] = 'F3';
        break;
 
      case 'cover':
        this.analyticsEvent.data[attributes.listEanPosition] = data.listEanPosition;
        this.analyticsEvent.data[attributes.listTitle] = data.listTitle;
        this.analyticsEvent.data[attributes.listViewAllClicked] = 'No';
        this.analyticsEvent.data[attributes.listPosition] = data.listPosition+1;
        this.analyticsEvent.data[attributes.shopSection] = 'F4';
        this.sendAnalytics(); // The componented doesn't unmount; call manually
        break;
 
      case 'list':
        this.analyticsEvent.data[attributes.listTitle] = data.listTitle;
        this.analyticsEvent.data[attributes.listViewAllClicked] = 'Yes';
        this.analyticsEvent.data[attributes.listPosition] = data.listPosition+1;
        this.analyticsEvent.data[attributes.shopSection] = 'F4';
        break;
      default:
        console.error('Invalid analytics type. What did you do wrong??');
    }
  }
 
 
  sendAnalytics() {
    
    // Send the event
    analytics.exit(this.analyticsEvent);
 
    // Reset the event data since we'll come back from PDP
    this.initAnalytics();
  }
