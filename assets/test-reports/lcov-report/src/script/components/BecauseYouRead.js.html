import React, { Component } from 'react'
import ListCarousel from './ListCarousel'
 
import '../../style-dist/components/BecauseYouRead.css'
 
 
class BecauseYouRead extends Component {
 
  renderCarousel(list) {
    const props = this.props;
 
    // This carousel is not a real list. We have to build a fake list when the
    // user opens it.
    return <ListCarousel
      index={ 1 }
      list={ list }
      eanData={ props.eanData }
      buildF1List={ this.props.buildF1List }
      onCoverClick={ props.onCoverClick }
      onListClick={ props.onListClick } />;
  }
 
 
  render() {
    let carousel = <div/>;
    if (this.props.isFetching) {
      carousel = <div />;
    }
 
    const data = this.props.data;
    if (data && data.id) {
      const carouselData = {
        title: "Recommended For You",
        listCloudId: data.id,
        eans: data.eans
      };
      carousel = this.renderCarousel(carouselData);
    }
 
    if (this.props.isFetching) {
      carousel = <div className="no-op" />;
    }
 
    return <div className="because-you-read f4-lists">
      { carousel }
    </div>;
  }
}
 
export default BecauseYouRead;
