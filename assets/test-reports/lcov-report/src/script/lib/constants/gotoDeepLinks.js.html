/**
 * Shop will be passed a URL param called 'goto' which will match one of the 
 * following objects. We should then change the URL hash to the appropriate path.
 */
 
 
const changeHash = hash => {
  const plat = window.Device.shopPlatform;
  window.location.hash = `/${plat}${hash}`;
};
 
 
const gotoDeepLinks = [
  {
    name: 'list',
    match: /\/store\/lists\/[0-9]+/,
    goto: query => {
      const listId = query.split('/')[3];
      changeHash(`/list/${listId}`);
    }
  },
  {
    name: 'nook channels',
    match: /\/store\/channels/,
    goto: query => {
      changeHash(`/channels/nkbk/affinity`);
    }
  },
  {
    name: 'category dref',
    match:/\/store\/.*\/[0-9]+/,
    goto: query => {
      const deptId = query.split('/')[2];
      const category =  query.split('/')[3];
      changeHash(`/home/categories/${deptId}/${category}`);
    }
  },
  {
    name: 'list of lists',
    match: /\/store\/listoflists/,
    goto: (query, params={}) => {
      let newHash = `/listoflists/${params.listIds}`;
      if (params.title) {
        newHash += `/${params.title}`;
      }
      changeHash(newHash);
    }
  }
];
 
 
export default gotoDeepLinks;
