import STRINGS_EN from "./en";
 
 
const getKeys = () => {
  let keyObject = {};
  let keys = Object.keys(STRINGS_EN);
  for (let key in keys) {
    keyObject[keys[key]] = keys[key];
  };
  return keyObject;
}
const KEY = getKeys();
 
export default KEY;
