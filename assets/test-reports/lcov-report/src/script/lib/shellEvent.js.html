class ShellEvent {
  init(options = {}) {
    return (evt) => {
      switch (evt) {
        case 'activate':
          options.toggleAppActive(true);
          break;
        case 'deactivate':
          options.toggleAppActive(false);
          break;
        case 'inStore':
          options.toggleInStore(true);
          break;
        case 'notInStore':
          options.toggleInStore(false);
          break;
        case 'back':
 
          const didNav = checkSpecialCases();
          if (didNav) {
            return;
          }
 
          // Close Shop if the back button is pressed from the for me tab
          if (window.location.hash.match('for_me')) {
            return window.Device.close();
          }
 
          // If we deep linked here, return to the landing
          if (window.history.length < 2) {
            return goToForMe();
          }
 
          window.history.back();
          break;
        case "purchase": {
          {
              let savedHash = window.sessionStorage.getItem("hash_before_entering_pdp");
              let currentHash = window.location.hash;
 
              if(savedHash === currentHash && window.listCover) {
                window.sessionStorage.removeItem("hash_before_entering_pdp")
                window.listCover.setPurchaseFlag();
              }
              return;
          }
        }
        default:
          return window.location.reload();
      }
    }
  }
 
}
 
 
function goToForMe() {
  const hash = window.location.hash.split('/');
  hash.length = 2;
  return window.location.hash = hash.join('/') + '/home/for_me';
}
 
 
// Return TRUE if special cases match
function checkSpecialCases() {
 
  // If we deep link to the list of Serial Reads titles from Readouts, go back to Serial Reads
  // NOTE: deep linking to serial reads was not possible before 5.3.0 native Android/iOS
  if (window.location.hash.match('list/10000012856')) {
    const appVersion = window.APP_VERSION;
 
    if (appVersion < "5.3.0") {
      goToForMe();
    }
    else {
      window.Device.launchUrl({url: 'http://www.nook.com/app/qr/serialreads'})
    }
    return true;
  }
 
  return false;
}
 
export default new ShellEvent();
