import actiontypes from '../lib/constants/actionTypes'
 
const initialState = {};
 
const favCategoriesData = (state = initialState, action) => {  
  let stateCopy = Object.assign({}, state);
 
  switch (action.type) {
 
    case actiontypes.SET_FAV_CATEGORIES_DATA:
      stateCopy.favCategories = action.favCategoriesData;
      return stateCopy;
 
    case actiontypes.UPDATE_FAV_CATEGORIES_DATA:
        stateCopy.favCategories = action.favCategoriesData;
        return stateCopy;
    
    default:
      return state;
  }}
 
export default favCategoriesData;
