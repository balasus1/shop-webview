import React, {Component} from 'react'
 
import '../../style-dist/components/EpdHomeOffline.css'
import textReplace from '../lib/i18n';
import KEY from '../lib/constants/languages/keys';
 
 
//
// Extend the EpdHomeOffline component to allow for easy default image loading
// 
class EpdHomeOffline extends Component {
 
  constructor(props) {
    super(props);
 
  }
 
  render() {
    if (window.navigator.onLine) {
      return <div />;
    }
 
    return <div className="epd-home-offline">
      <div className="title">
        {textReplace(KEY.you_are_offline)}
      </div>
 
      <div className="sub-title">
        {textReplace(KEY.no_internet)}
      </div>
    </div>;
  }
}
 
export default EpdHomeOffline
