import actions from '../lib/constants/actionTypes'
 
const initialState = {
  categories: {},
  staticMap: {},
  drefHash: {
    // 1607: {id: "1607", title: "Kid's Books", childCount: 2},
    // 3158: {id: "3158", title: "eNewsPapers", childCount: 2},
    // 3159: {id: "3159", title: "eMagazines", childCount: 2},
    // 21954: {id: "21954", title: "Digital Audiobooks", childCount: 2},
    // 21955: {id: "21955", title: "Kid's AudioBooks", childCount: 2},
  }
};
 
const categoryData = (state = initialState, action) => {
  let stateCopy = Object.assign({}, state);
  let catCopy = Object.assign({}, state.categories);
 
  switch (action.type) {
    case actions.SET_CATEGORY_DATA:
      let catStore = {};
      catStore[action.code] = {
        refinements: action.data.refinements,
        parent: action.data.parent || null,
        spots: action.data.spots || null
      };
 
      Object.assign(catCopy, catStore);
      stateCopy.categories = catCopy;
      return stateCopy;
 
    // This is a hash of category names that we will keep as we fetch
    // categories.
    case actions.SET_DREF_HASH:
      const hash = {};
 
      if (action.data && action.data) {
        for (let key in action.data.allCategories) {
          let cat = action.data.allCategories[key]
          hash[cat.id] = cat;
        }
        for (let key in action.data.topCategories) {
          let cat = action.data.topCategories[key]
          if (!hash[cat.id]) {
            hash[cat.id] = cat;
          }
        }
      }
 
      Object.assign(stateCopy.drefHash, hash);
      return stateCopy;
      
    // This is a static map that contains (nearly) everything
    // ONLY USE THIS FOR THE RECOMMENDED CATEGORY "PODS"
    // case actions.SET_STATIC_DREF_MAP:
    //   stateCopy.staticMap = action.data;
    //   return stateCopy;
    
    default:
      return state;
  }
}
 
export default categoryData
