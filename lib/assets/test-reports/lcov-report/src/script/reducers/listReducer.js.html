import actions from '../lib/constants/actionTypes';
 
const initialState = {
  listData: {
    loadedListId: '',
    eans: []
  },
  listDetails: {
    listId: '',
    listTotal: '',
    listTitle: '',
    listDescription: '',
    listTnc: {
      link: '',
      message: ''
    }
  }
};
 
const ListData = (state = initialState, action) => {
  let stateCopy = Object.assign({}, state);
 
  switch (action.type) {
 
    case actions.SET_LIST_DETAILS:
      return Object.assign(stateCopy, {
        listDetails: {
          listId: action.listId,
          listTotal: action.listTotal,
          listTitle: action.listTitle,
          listDescription: action.listDescription,
          listTnc: action.listTnc
        }
      });
 
    case actions.SET_LIST_ITEMS:
      stateCopy.listData = action.listData;
      return stateCopy;
    
    case actions.ADD_LIST_ITEMS:
      let existingData = stateCopy.listData;
      const listDataCopy = Object.assign({}, stateCopy.listData);
      listDataCopy.eans = existingData.eans.concat(action.listData.eans);
      stateCopy.listData = listDataCopy;
      return stateCopy;
 
    default:
      return state;
  }
}
 
export default ListData
