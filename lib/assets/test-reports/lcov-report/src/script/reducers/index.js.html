//
// Each reducer listens for an action "type" from a dispatch event, 
// then modifies the application state (redux store) using a "pure funciton".
// 
import { combineReducers } from 'redux'
import sessionReducer from './sessionReducer'
import shellReducer from './shellReducer'
import configReducer from './configReducer'
import affinityReducer from './affinityReducer'
import eanReducer from './eanReducer'
import forMeReducer from './forMeReducer'
import categoryReducer from './categoryReducer'
import listReducer from './listReducer'
import channelsReducer from './channelsReducer'
import lolReducer from './lolReducer'
import epdReducer from './epdReducer'
import epdHomeReducer from './epdHomeReducer'
import favoriteCategoriesReducer from './favoriteCategoriesReducer'
 
// 
// Build the redux reducer by combining each child reducer.
// DOCUMENTATION:
//   http://redux.js.org/docs/api/combineReducers.html
// 
const shopReducer = combineReducers({
  configData: configReducer,
  sessionData: sessionReducer,
  affinityData: affinityReducer,
  eanData: eanReducer,
  forMeData: forMeReducer,
  favCategoriesData: favoriteCategoriesReducer,
  categoryData: categoryReducer,
  listData: listReducer,
  lolData: lolReducer,
  channelsData: channelsReducer,
  epdData: epdReducer,
  shellData: shellReducer,
  epdHomeData: epdHomeReducer
});
 
export default shopReducer;
