  loadCategories(toFetch, options) {
 
    this.setState({ isFetchingCategories: true });
 
    // Callback function - fetch complete
    const done = (err, response) => {
      if (err) {
        console.error('Error fetching category data.', err.message);
        window.Device.showError('Error fetching category data. ' + err);
        response = { eans: [], categories: [], links: [] };
      }
      else {
        this.setState({ isFetchingCategories: false });
      }
      let isAudiobook = !!window.location.href.match(constants.AUDIOBOOK.toLowerCase()) || window.SHOW_AUDIOBOOKS;
      let categories = getCurrentCategories(response, this.props.childProfile, isAudiobook);
      
      this.setState({ 'categories': categories});
      const catData = {
        data: categories,
      };
 
      this.props.setStoreCategories(catData, toFetch.storeCode); // Trigger a redux dispatch
    };
    fetchStoreCategories(toFetch.storeCode, {}, done); // Cached for 12 hrs
  }

      window.paginationComponent.gotoPage(1);

    }
  }

  setAllFavsHeight(e) {
    if (window.allCats && e.detail.categoryType === constants.ALL_FAVS) {
      let landingHeight = window.location.href.match(platforms.epd) ? 492 : 704;
      window.allCats.style.height = (landingHeight * e.detail.page) + "px";
      window.allCats.scrollTop = (e.detail.page - 1) * landingHeight;
    } else if(window.favInProfile) {
      const wrapper = document.getElementById('shop-wrapper');
      window.favInProfileContainer.style.height = ((wrapper.clientHeight - 50) * e.detail.page) + "px";
      wrapper.scrollTop = (e.detail.page - 1) * (wrapper.clientHeight - 100);
    }

  loadIfNoData(props) {

      this.loadCategories({

      load()

        this.props.setStoreCategories(catData, store.code); // Trigger a redux dispatch
    }

  renderSubCategories(code, categories, store) {
    // Show a list view for L2 categories
    // In the case of a synthetic category (kids, newsstand, etc.) we want to
    // show this view for the L2, and go to list view for L3
    // Removed above logic based on SHOP-442
    const showIcon = this.state.showCategoriesList ? listIcon : gridIcon;
    const topCategoriesIcon = this.state.showTopCategories ? expandIcon : collapseIcon;
    const allCategoriesIcon = this.state.showAllCategories ? expandIcon : collapseIcon;
 
    const toggleList = this.state.showCategoriesList === false ? this.openList : this.closeList;
    const toggleTopCategories = () => this.setState({ showTopCategories: !this.state.showTopCategories });
    const toggleAllCategories = () => this.setState({ showAllCategories: !this.state.showAllCategories });
 
    // 6.4 phase 2 API change
    // To Do: Replace this temp with Top Categories API response.
    /* ******************************************************** */
    //let temp = categories[code].refinements.slice(0, 12);
 
    let cats = this.state.isFetchingCategories ? {topCategories:{}, allCategories:{}} : this.state.categories;
    let topCats = [];
    cats.topCategories =  this.state.isFetchingCategories ? this.dummyCats : reOrder(cats.topCategories);
 
    if (cats && cats.topCategories) {
      let index = 0;
      for (let key in cats.topCategories) {
        if (index > 11) {
          break; // Max 12 categories are allowed in UI
        }
        let cat = cats.topCategories[key]
        topCats.push(<CategoryLink key={cat.id}
          showList={true}
          storeCode={store.code}
          cat={cat}
          name={cat.label}
          dref={cat.id}
          className="l2-links"
          showIcon={true}
          showCategoriesList={this.state.showCategoriesList}
          childProfile = {this.props.childProfile}
          cats={this.state.categories} />);
          index++;
      }
    }
    let allCats = [];
 
    let parentCats = getParentsFromAllCategories(cats.allCategories);
 
    parentCats = this.state.isFetchingCategories ? this.dummyCats : reOrder(parentCats);
 
    for (let key in parentCats) {
      let cat = parentCats[key]
      allCats.push(<CategoryLink key={cat.id}
        showList={true}
        storeCode={store.code}
        cat={cat}
        name={cat.label}
        dref={cat.id}
        className="l2-links"
        showIcon={false}
        showCategoriesList={this.state.showCategoriesList}
        childProfile = {this.props.childProfile}
        cats={this.state.categories} />)
    }
 
    const cat = this.props.drefHash[String(code)];
    const title = cat ? cat.title : store.name; // Default in case something goes wrong
    return <div key={code} ref={el => window.subCategories = el} className={"sub-categories " + (this.state.showCategoriesList ? "grid" : "")}>
      <div className="explore-link">
        {
          // Don't show the "Explore" link for "departments" that aren't actually categories
          cat && <CategoryLink key={cat.id}
            showList={true}
            storeCode={store.code}
            name={'See All ' + title}
            dref={cat ? (cat.id == null ? this.props.currentDref : cat.id) : store.code}
            childProfile = {this.props.childProfile} />
          //dref={ cat ? cat.id : store.code } />
        }
      </div>
      <div className='categories-title browse'>
        <span className='browse-categories-title'>
         {textReplace(KEY.browse_categories)}
        </span>
        <a onClick={toggleList}>
        <img src={showIcon} alt={textReplace(KEY.list_icon)} className="list-icon"></img>
        </a>
      </div>
 
      <div className='categories-title top'>
        <span>
        {textReplace(KEY.top_categories)}
        </span>
        {!this.isEPD && <img src={topCategoriesIcon} alt={textReplace(KEY.top_categories_icon)} className="list-icon"

      </div>

      <div className= {'all-categories-container'  + (this.state.showAllCategories ? " show" : " hide") }>
        {allCats}
      </div>
 
    </div>;
  }
 
  renderFavAllCategories(code, categories, store) {
 
    let cats = this.state.categories;
    let parentCats = null;
    let indexToRemove = 0;
    let allCats = [];
    parentCats = getParentsFromAllCategories(cats.allCategories);
    parentCats = this.state.isFetchingCategories ? this.dummyCats : reOrder(parentCats);
 
    for (let key in parentCats) {
      let cat = parentCats[key];
      if (this.props.selectedFavCategories) {
        indexToRemove = this.props.selectedFavCategories.findIndex(cat2Remove => {
          return cat2Remove.label === cat.label;
        });
        if (indexToRemove > -1) { // It's in favorites. Don't push to All Categories
          continue;
        }
      }
      allCats.push(<CategoryLink key={cat.id}
                                 showList={true}
                                 isInFavoriteCategoriesPage={ this.props.isInFavoriteCategoriesPage }
                                 storeCode={store.code}
                                 cat={cat}
                                 name={cat.label}
                                 dref={cat.id}
                                 className="l2-links"
                                 isCategorySelected={ () => this.isCategorySelected }
                                 showIcon={ !!this.props.isInFavoriteCategoriesPage }
                                 addOrRemoveItem={ (e) => {
                                   this.props.addOrRemoveItem(e);
                                 } }
                                 showCategoriesList={this.state.showCategoriesList}
                                 childProfile = {this.props.childProfile}
                                 cats={this.state.categories}  />)
    }
 
    if (this.isEPD && !window.favInProfile) {
      let noOfFillers = 10 - (allCats.length % 10);
      for (let i = 0; (i < noOfFillers && noOfFillers < 10); i++) {
        allCats.push(<div key={i + "filler"} className="category-link l2-links fav-cat-section filler"></div>)
      }
    }
 
    return <div key={code} className="sub-categories">
      <div className='all-categories-container'  ref={el => window.allCats = el}>
        {allCats}
      </div>
    </div>;
  }
 
  openList() {
 
    if (this.state.showCategoriesList === false) {
      this.setState({ showCategoriesList: true });
    }
 
    sessionStorage.setItem('showCategoriesList', 'grid');
  }
 
 
  closeList() {
    this.setState({ showCategoriesList: false });
    sessionStorage.setItem('showCategoriesList', 'list');
  }
 
  expandAllCategories() {
    this.setState({ showAllCategories: true });
  }
 
  collapseAllCategories() {
    this.setState({ showAllCategories: false });
  }
 
 
  render() {
    let loading = this.state.isFetchingCategories;
    const store = this.props.currentStore;      // Deep link store
    const currentDref = this.props.currentDref; // Deep link DREF
    const loadingAriaAttribs = {
      "aria-busy": true, "aria-live": "off"
    };
    // Check for loaded data
    const code = currentDref || store.dref || store.code;
    if(this.state.isFetchingCategories) {
      this.state.categories = this.dummyCats;
    }
    let storeCats = this.state.categories && !this.props.isInFavoriteCategoriesPage ?
        this.renderSubCategories(code, this.state.categories, store) :
        this.renderFavAllCategories(code, this.state.categories, store);
 
    return <div {...(loading ? loadingAriaAttribs : { "aria-live": "off" })} 
    className={'store-categories' + (loading ? ' loading' : '') + (this.isiOS ? " ios" : "")}>
      {storeCats}
    </div>;
  }
}
