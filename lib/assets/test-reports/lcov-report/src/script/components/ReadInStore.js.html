import React, { Component } from 'react'
import KEY from '../lib/constants/languages/keys';
import textReplace from '../lib/i18n';
 
class ReadInStore extends Component {
 
	render() {
		if (this.props.inStore) {
			return <div className={"read-in-store"}><span>{textReplace(KEY.read_in_store)}</span></div>;
		}
		else {
			return <div></div>;
		}
	}
 
}
 
export default ReadInStore
