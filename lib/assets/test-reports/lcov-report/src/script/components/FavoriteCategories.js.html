class FavoriteCategories extends Component {
  constructor(props) {
    super(props);
    this.shopType = this.props.match.params.shopType;
    window.favInProfile = !!this.props.match.params.shopType
    this.isEPD = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
    this.isEpd = window.location.href.match(platforms.epd);
    this.isIOS = window.location.href.match(platforms.ios);
    this.state = {
      isTab: (!this.shopType || this.isEPD) && window.innerWidth > 480,
      isChecked: true,
      selectedFavCategories: [],
      saving: false,
      displayBackIcon: false
    };
 
    this.categoryStoreCode = 'nkbk';
    this.shopHome = (this.shopType &&
      ((this.shopType === textReplace(KEY.audiobook).toLowerCase()) ? "audiobook" : "ebook")) //invoked from profile screen
      || (window.SHOW_AUDIOBOOKS ? "audiobook" : "ebook"); //Not invoked from profile
    this.isInFavoriteCategoriesPage = !!window.location.href.match("favCategories");
    // bind methods
    this.selectFavCategories = this.selectFavCategories.bind(this);
    this.closeFavCategories = this.closeFavCategories.bind(this);
    this.setSelectedFavsHeight = this.setSelectedFavsHeight.bind(this);
    this.isOrientationLandscape = this.isOrientationLandscape.bind(this);
    this.isEPD = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
    this.key = `selectedFavCategories_${this.props.customer.accountId}-${this.props.customer.profileId}-${this.shopHome}`;
    this.blurEvent = window.location.href.match(platforms.ios) ? "blur" : "visibilitychange";
    this.initialLoad = true;
	}
 
  componentDidMount() {
 
    const showBackIcon = () => {
      let displayBackIcon = !(window.screen.width < 480)  // not phone
 
      if (window.screen.width < 480 || window.screen.height < 480) {
        displayBackIcon = false;
      }
 
      else if(window.screen.width >= 480 && window.innerWidth <= Math.ceil(.4 * window.screen.width)) { // splitscreen 
        displayBackIcon = true;
      }
 
      this.setState({ displayBackIcon });
 
    }
 
    if (this.isIOS) {
      window.addEventListener('resize', this.isOrientationLandscape);
      if (window.favInProfile) {
        this.isOrientationLandscape();
      }
    }
    else {
      window.addEventListener("resize", () => {
        this.setState({ isTab: (!this.shopType || this.isEPD) && window.innerWidth > 480 });
        if (window.favInProfile) {
          showBackIcon();
        }
      });
      if (window.favInProfile) {
        showBackIcon();
      }
    }
 
    window.scroll(0,0); //start from top of the page.
    let interval = setInterval(() => {
      if (document.readyState === 'complete') {
        clearInterval(interval);
        window.scroll(0, 0);
      }
    }, 300);
 
    const preserveSelection = () => {
      let customer = this.props.customer;
      let urlParams = {
        accountId: customer.accountId,
        profileId: customer.profileId,
        shopHome: this.shopHome
      };
 
      if (sessionStorage.getItem('reloaded') != null) { // Retain selected favs on refresh.
        console.log("This Screen is reloaded...Retaining selected cats..")
        let value = localStorage.getItem(this.key);
        this.state.selectedFavCategories = JSON.parse(value)
        this.saveFavCategories(true);
        setTimeout(() => {
          this.loaded = true;
        }, 1000)
      }
      else {
        this.loadFavCategories(urlParams);
      }
    }
 
    window.addEventListener('onScrollToPage', this.setSelectedFavsHeight);
 
    window.addEventListener(this.blurEvent, this.saveOnBlur());
 
    preserveSelection();
  }
 
  saveOnBlur() {
    if (!this.initialLoad && performance.navigation.type !== 1) { // Retain selected favs on refresh.
      this.closeFavCategories(true);
    }
 
    this.initialLoad = false; // Do not close on initial load. 
 
    if (this.shopType && window.screen.width <= 480) {
      window.Device.modifyHeader({
        enableBack: true,
        title: constants.FAV_CATEGORIES,
        hideTools: true,
        hideBottomNav: false
      });
    }
  }
 
 
  setSelectedFavsHeight(e) {
    if (!this.shopType && window.selectedCats && e.detail.categoryType === constants.SELECTED_FAVS) {
      let isEPD = window.location.href.match(platforms.epd);
      let landingHeight = isEPD ? 390 : 603;
      window.selectedCats.style.height = (landingHeight * e.detail.page) + "px";
      window.selectedCats.scrollTop = (e.detail.page - 1) * (landingHeight);
    }
  }
 
  isOrientationLandscape() {
    window.Device.isOrientationLandscape(response => {
      const responseObj = (typeof response === 'object') ? response : JSON.parse(response);
      if (responseObj) {
        const innerWidth = parseInt(JSON.parse(responseObj.innerWidth));
        if (window.favInProfile) {
          this.setState({ displayBackIcon: !JSON.parse(responseObj.splitscreen)});
        } else if (innerWidth > 480) {
          this.setState({ isTab: !responseObj.orientation ? true : responseObj.orientation });
        } else if (!window.favInProfile && innerWidth < 480) {
          this.setState({isTab: false});
        }
      }
    });
  }
 
  componentDidUpdate(prevProps, prevState) {
    if(this.isIOS && (prevState.isTab && (prevState.isTab !== this.state.isTab))) {
      this.isOrientationLandscape();
    } else if(this.isIOS && (!prevState.isTab && (prevState.isTab !== this.state.isTab))) {
      this.isOrientationLandscape();
    }
  }
 
  componentWillUnmount() {
    this.initialLoad = false;
    window.removeEventListener(this.blurEvent, this.saveOnBlur());
    window.removeEventListener('resize', this.isOrientationLandscape);
  }
 
  loadFavCategories(urlParams) {
    const props = this.props;
    favoriteCategories('get', urlParams, null, (err, resp)  => {
      this.loaded = true;
      if(err) {
        console.error('Error fetching Favorite Categories.');
        resp = {favorites:[]};
      }
      let responseData = this.shopHome === "ebook" ? resp.data.ebook : resp.data.audiobook
      if(!responseData) { // Remove this check once NOOKCLOUD-5665 goes to prod
        responseData = resp.data.favorites;
      }
      this.setState({selectedFavCategories: responseData});
      props.setFavCategories(responseData); // Trigger a redux dispatch
    });
  }
 
  async saveFavCategories(noRedirect) {
    let customer = this.props.customer;
    let oldFavCategories = this.props.data.favCategories;
    let newlySelectedFavCategories = this.state.selectedFavCategories;
    let urlParams = {
      accountId: customer.accountId,
      profileId: customer.profileId,
      shopHome: this.shopHome
    };
    let payLoad = {
      "favorites": newlySelectedFavCategories
    }
 
    const redirect = (err) => {
      if(noRedirect) {return;}
      if (err) {
        console.error('Error saving Favorite Categories.' + err);
        this.props.updateFavCategories(oldFavCategories);
        return;
      }
      else {
        if (this.shopType && window.location.hash.match(`/favCategories/${this.shopType}`)) { //shop type will be there when navigated from profile.
            window.history.back();
        }
        else if (!this.shopType && !window.location.hash.match('/favupdate')) {
        localStorage.setItem(this.key, JSON.stringify(newlySelectedFavCategories));
        if (window.Device.shopPlatform !== platforms.ios && !window.LOCALHOST) {//deeplink to for me screen as the other redirection method works only for IOS.
          window.Device.launchUrl({ url: 'http://www.nook.com/app/' + (window.SHOW_AUDIOBOOKS ? "audiobookstore" : "store") });
        } else {
          window.location.hash = window.location.hash.replace("/favCategories", "/for_me/favupdate");
        }
      }
    }
  }
 
    //Check if any changes made.
    if (newlySelectedFavCategories && oldFavCategories && !isSameObject(newlySelectedFavCategories, oldFavCategories)) {
      if (!noRedirect) {
        window.Device.showToast({
          message: textReplace(KEY.saving_favorites)
        });
      }
    } else if (this.shopType) {
      redirect();
    }
 
    localStorage.setItem(this.key, JSON.stringify(newlySelectedFavCategories));
    this.props.updateFavCategories(newlySelectedFavCategories);
 
    if (!window.LOCALHOST) {
      payLoad = {
        shopType: urlParams.shopHome,
        categories: newlySelectedFavCategories,
        closeModal: !noRedirect
      }
      try {
        postFavoriteCategoriesNative(payLoad);
      } catch (e) {
        console.log('Error posting Favorite Categories to native: ' + e);
      }
      redirect();
    } else {
      favoriteCategories('post', urlParams, payLoad, (err, resp) => {
        redirect(err);
      });
    }
  }
 
  renderFavorite(eachCategory, i) {
    return (
      <div key={eachCategory.id} className="fav-category">
        <img src={ this.state.categoryImage[eachCategory.id] } className="category-icon" alt={ eachCategory.label } />
        <div className={'category-name ellipses'}>{  this.state.categoryLabel[eachCategory.id] || eachCategory.label }</div>
        <form>
          <input
              type="image"
              src={ checkedBox }
              alt={textReplace(KEY.submit)}
              className="fav-selection-checkbox"
              id={ eachCategory.id }
              name={ eachCategory.label }
              defaultChecked={ this.state.isChecked }
              value={ eachCategory.label }
              onClick={ this.selectFavCategories }
          />
        </form>
 
      </div>
    );
  }
 
  renderFiller() {
    let maxSelected = this.shopType ? 10 : 9;
    let noOfFillers = maxSelected - (this.state.selectedFavCategories.length % maxSelected);
    let fillers = [];
    for (let i = 0; (i < noOfFillers && noOfFillers < maxSelected); i++) {
      fillers.push(<div key={i + "filler"} className="fav-category filler"></div>);
    }
    return fillers;
  }
 
  // select only checked favorite categories and store in array
  selectFavCategories(event) {
    let updatedItems = [...this.state.selectedFavCategories];
    const isIndexFound = updatedItems.findIndex(selectedCategory => {
    return selectedCategory.label === event.target.name;
    });
    if (event.target.checked) {
      if(isIndexFound === -1) {
        updatedItems.push({"id": event.target.id, "label": event.target.value});
      } else {
        updatedItems.splice(isIndexFound, 1);
      }
    } else {
      if (isIndexFound > -1) {
        updatedItems.splice(isIndexFound, 1);
      } else {
        updatedItems.push({"id": event.target.id, "label": event.target.value});
      }
    }
    const ordered = reOrder(updatedItems, 'label');
    this.setState({selectedFavCategories: ordered});
    localStorage.setItem(this.key, JSON.stringify(ordered));
    
    if (updatedItems && updatedItems.length === 0 && window.selectedCats) {
      window.selectedCats.removeAttribute("style");
    }
    else {
      sessionStorage.setItem('reloaded', "true");
    }
    this.initialLoad = false;
  }
 
  closeFavCategories = () => {
    try {
      if (this.loaded) {
        this.saveFavCategories();
      }
      sessionStorage.removeItem('reloaded');
    } catch (e) {
      console.log(`Error while saving favorite categories: ${e}`);
    }
  }

                </div>
              </div> :
              <React.Fragment>
               { isCategoriesSelected && <div className="selected-fav-categories">
                  { this.state.isTab && !this.shopType && favCategoryInstruction }
                  <div className="title">{ constants.FAV_CATEGORIES }</div>
                  { this.state.selectedFavCategories && this.state.selectedFavCategories.map(category => {
                    return this.renderFavorite(category, category.id);
                  }) }
                </div> }
              <div className={"selected-fav-categories all-categories" + ((this.shopType && !this.isEPD)? " profile" : "") }>
              <div className="title">{ constants.ALL_CATEGORIES }</div>
              <CategoriesContainer addOrRemoveItem={ (e) => { this.selectFavCategories(e);} }

  }
