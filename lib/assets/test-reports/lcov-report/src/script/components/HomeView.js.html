import React, { Component } from 'react'
import { Route } from 'react-router-dom'
import viewTypes from '../lib/constants/viewTypes'
import platforms from '../lib/constants/platforms'
 
import HomeTabs from '../components/HomeTabs'
import ForMeContainer from '../containers/ForMeContainer'
import CategoriesContainer from '../containers/CategoriesContainer'
import CategoryListContainer from '../containers/CategoryListContainer'
 
import '../../style-dist/components/HomeView.css'
 
 
class HomeView extends Component {
 
  render() {
    const chosenTab = this.props.match.params.tab;
    const match = this.props.match;
    const showDevTabs = false;
 
    return (
      <section className={ 'shop-view shop-home' + (showDevTabs ? ' dev' : '') }>
        { 
          // Show "mock" top-nav button in dev to make life easier
          showDevTabs && <HomeTabs chosenTab={ chosenTab } />
        }
        <div className="tab-children">
          <Route exact path={ `${match.url}/${viewTypes.home.FOR_ME}` } component={ ForMeContainer } />
          <Route exact path={ `${match.url}/${viewTypes.home.FOR_ME}/favupdate` } component={ ForMeContainer } />
          {/* <Route exact path={ `${match.url}/${viewTypes.home.CATEGORIES}/:catStoreCode/:catDref/list` } component={ CategoryListContainer } /> */}
          <Route exact path={ `${match.url}/${viewTypes.home.CATEGORIES}/:catStoreCode/:catDref/list/:filterSort?` } component={ CategoryListContainer } />
          {/* <Route exact path={ `${match.url}/${viewTypes.home.CATEGORIES}/:catStoreCode/:catDref/list/sort` } component={ CategoryListContainer } /> */}
          <Route exact path={ `${match.url}/${viewTypes.home.CATEGORIES}/:catStoreCode/:catDref` } component={ CategoriesContainer } />
          <Route exact path={ `${match.url}/${viewTypes.home.CATEGORIES}/:catStoreCode` } component={ CategoriesContainer } />
          <Route exact path={ `${match.url}/${viewTypes.home.CATEGORIES}` } component={ CategoriesContainer } />
        </div>
      </section>
    );
  }
}
 
export default HomeView;
