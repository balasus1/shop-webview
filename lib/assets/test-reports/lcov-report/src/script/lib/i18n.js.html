const textReplace = key => {
  let lang = (navigator.language || 'en-US').replace(/-.*/g, '');
  lang = window.location.href.match(platforms.ios) ? lang : "en"; //es is supported only for iOS
 
  let strings = '';
  switch(lang) {
    case 'en':
      strings = en;
      break;
    case 'es':
      strings = es;
      break;
    default:
      strings = en;
