// Customer Affinities
// 
// Affinities are cached for 12 hours since they rarely change.
// 
export const fetchAffinities = (userParams, limit, callback) => {
  let paramString = paramsFromObj(userParams);
  const good = (response) => {
    let affinities = [];
    if (response.UserProfile && response.UserProfile.Affinities && response.UserProfile.Affinities.length) {
      affinities = response.UserProfile.Affinities[0].Records || [];
    }
 
    // Set default affinities to make things work nicely
    // Note: ebook affinities are used for both bookstore and audiobooks as fallback
    if (!affinities.length) {
      affinities = affinityDefaults.slice();
    }
 
    if(affinities) {
      affinities = affinities.slice(0, limit);
    }
    callback(null, affinities);
    window.cache.set(`${userParams.storeCode}_affinity-data`, affinities, 12);
  };
 
  const bad = (err) => {
    console.error('Affinity fetch failed');
    callback(true);
  };
 
  const cached = window.cache.get(`${userParams.storeCode}_affinity-data` );
  if (cached) {
    return callback(null, cached);
  }
 
  return assemble('get', '/proxy/affinity', paramString, null, good, bad);
}

// F1 Recommendations (Originally from Nook Home)
// These are based on the user's recent Library items
// 
export const fetchF1 = (userParams, callback, options={}) => {
 
  const doFetch = (recentItems) => {
    // Build the object of items and ages
    const age = 60 * 60 * 6; // We will approximate the age since we don't know it
    const recentEans = Object.keys(recentItems);
 
    const shuffle = (recentEans) => {
      let counter = recentEans.length;
      // While there are elements in the recentEans
      while (counter > 0) {
        // Pick a random index
        let index = Math.floor(Math.random() * counter);
        // Decrease counter by 1
        counter--;
        // And swap the last element with it
        let temp = recentEans[counter];
        recentEans[counter] = recentEans[index];
        recentEans[index] = temp;
      }
      return recentEans;
    }
 
    const getRandomValues = (recentEans, count) => {
      count = recentEans.length < count ? recentEans.length : count;
      return shuffle(recentEans).slice(0, count);
    }
 
    const eans = getRandomValues(recentEans, 6); //Get 6 random eans for recommendations
 
    const items = eans.map((ean, idx) => {
      return {
        ean,
        ageInSeconds: age * (idx + 1) * 3
      };
    });
 
    const params = Object.assign({
      recentlyOpened: JSON.stringify(items),
      size: options.size || 16 // Straight from Maria
    }, staticParams());
    let paramString = `${paramsFromObj(params)}&${paramsFromObj(userParams)}`;
 
    const good = (response) => {
      if (!response) {
        return callback('No F1 data received');
      }
 
      // Get the recommended from recent activity pod (rec_from_ract)
      let pod = response.spots.f1.find(pod => pod.id === 'basedOnActivity');
      if (!pod) {
        pod = response.spots.f1[0]; // Use the first pod if the "basedOnActivity" pod isn't returned
      }
 
      if (pod && pod.data) {
        pod = pod.data.main;
      }
 
      if (response.eans) {
        let eanKeys = Object.keys(response.eans);
        for (let index = 0; index < eanKeys.length; index++) {
          if (window.Device.itemIsInLibrary(eanKeys[index])) {
            delete response.eans[eanKeys[index]];
          }
        }
      }
 
      const data = {
        recommendations: pod,
        eans: response.eans
      };
 
      callback(null, data);
 
      if (options.offlineCache) {
        window.cache.set(`${userParams.storeCode}_f1-data` , data);
      }
    };
 
    const bad = (err) => {
      console.error('Pod fetch failed');
      callback('Pod fetch failed');
    };
 
    // If there is cached data and we are offline, return the cached data
    if (options.offlineCache) {
      const cached = window.cache.get(`${userParams.storeCode}_f1-data`);
      if (cached) {
        return window.Device.isConnected(connected => {
          if (!connected) {
            callback(null, cached);
          }
          else {
            return assemble('get', '/home/pods/a10', paramString, null, good, bad);
          }
        });
      }
    }
 
    return assemble('get','/home/pods/a10', paramString, null, good, bad);
  }
 
  const error = () => {
    console.error('Error getting recentlyOpened.');
  }
 
  // We need to pass the user's recent library items to this call
  window.Device.getRecentLibraryItems(12, doFetch, error);
}

    }

      status: response.status.success
    }

  };
 
  const bad = (err) => {
    console.error('F4 lists fetch failed');

export const fetchStoreCategories = async (storeCode, userParams, callback) => {
 
  const params = {
    storeCode: storeCode,
    retailer: 'BN2',
    context_locale: 'en_US',
    includeEanDetails: true,
    profileType:"both"
  }
  let paramString = `${paramsFromObj(params)}&${paramsFromObj(userParams)}`;
  const cacheKey = 'categories';
  const good = (response) => {
    if (!response || !response.data) {
      return callback('No Store Categories received.');
    }
 
    const catData = {
      data: response.data
    };
 
    callback(null, catData);
    if (window.LOCALHOST) {
      window.db.setCategoriesToNative(cacheKey, catData);
      window.cache.set(`categories_expiry` , {expiry: hrsUntilMidnight()}, hrsUntilMidnight(), true);
    }
 
  };
 
  const bad = (err) => {
    console.error('Category fetch failed', err);
    callback(err);
  };
 
  const dbsuccess = (value) => {
    if (value) {
      console.log(`fetched from native-fetchStoreCategories`);
      return callback(null, typeof value === 'object' ? value : JSON.parse(value));
    }else {
      console.log(`fetched from cloud-fetchStoreCategories`);
      return assemble('get', '/categories/all', paramString, null, good, bad);
    }
  }
 
  const dbfailed = () => {
    console.log(`Error/dbfailed: Unable to fetch categories from native db`);
    return assemble('get', '/categories/all', paramString, null, good, bad);
  }
 
  window.db.getCategoriesFromNative(cacheKey, dbsuccess, dbfailed);
 
  if (window.LOCALHOST) {
    const cached = window.cache.get(`categories_expiry`, true);
    if (cached) {
      window.db.getCategoriesFromNative(cacheKey, dbsuccess, dbfailed);
    } else {
      return assemble('get', '/categories/all', paramString, null, good, bad);
    }
  }
}
