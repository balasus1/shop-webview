//
// Test the native functionality of a device.
//
import JsniAsync from '../../classes/JsniAsync'
 
const deviceTest = {
 
  test: () => {
 
    console.log('');
    console.log('TESTING JSNI...');
    if (window.JSNI.isMock) {
      console.warn('-- MOCK JSNI LOADED');
    }
 
    let jsniAsync = new JsniAsync();
    const $JSNI = jsniAsync.build$JSNI();
 
    // Test getSessionInfo
    $JSNI.getSessionInfo().then(
      sdata => {
        console.log('-- session info:');
        try {
          console.log(JSON.parse(sdata));
        } catch (e) {
          console.error('--', e);
        }
      },
      err => {
        console.error('-- error:', err);
      }
    );
 
    // Test isConnected
    $JSNI.isConnected().then(
      connected => {
        console.log('-- isConnected:');
        try {
          console.log(connected);
        } catch (e) {
          console.error('--', e);
        }
      },
      err => {
        console.error('-- error:', err);
      }
    );
 
    // Test getRecentLibraryItems
    window._testPass = (data) => {
      console.log('-- Pass');
      console.log(data);
    };
 
    window._testFail = (err) => {
      console.error('-- error', err);
    };
 
    console.log('');
    console.log('-- Library Items:')
    window.JSNI.getLibraryItems('_testPass', '_testFail');
  }
};
 
export default deviceTest
