const getParentsFromAllCategories = (allCategories) => {
  let parentCats = [];
  if (allCategories) {
    let keys = Object.keys(allCategories);
    for (let i = 0; i < keys.length; i++) {
      let record = allCategories[keys[i]];
      if (!record.parent) {
        parentCats.push(record);
      }
    }
  }
  return parentCats;
}
 
const reOrder = (unOrderedList, custom) => {

  if(custom === 'label') {
    ordered = temp.sort(function (a, b) {
      if (a.label < b.label) {
        return -1;
      }
      if (a.label > b.label) {
        return 1;
      }
      return 0;
    });

const getCssStyle = (element, prop) => {
  return window.getComputedStyle(element, null).getPropertyValue(prop);
}
 
const getCanvasFontSize = (el = document.body) => {
  const fontWeight = getCssStyle(el, 'font-weight') || 'normal';
  const fontSize = getCssStyle(el, 'font-size') || '16px';
  const fontFamily = getCssStyle(el, 'font-family') || 'Times New Roman';
 
  return `${fontWeight} ${fontSize} ${fontFamily}`;
}
 
const getElementPadding = (element) => {
  let leftPaddingValue = getCssStyle(element, 'padding-left') || '0px';
  leftPaddingValue = Number(leftPaddingValue.replace("px", ""));
  let rightPaddingValue = getCssStyle(element, 'padding-right') || '0px';
  rightPaddingValue = Number(rightPaddingValue.replace("px", ""));
  return leftPaddingValue + rightPaddingValue;
}

  let rightMarginValue = getCssStyle(element,'margin-right') || '0px';
  rightMarginValue = Number(rightMarginValue.replace("px", ""));

}
 
 
const addEllipsis = (text) => {
  let topCatsPadding = getElementPadding(text);
  let topCatsMargin = getElementMargin(text.parentNode.parentNode);
  let exploreCatsParentPadding = getElementPadding(text.parentNode);
  let isWrap = false;
 
  if (text) {
    let words = text.innerText.trim().split(" ");
 
    if (words.length > 1) {
      let word = text.innerText.trim().split(" ")[0];
      let textWidth = getTextWidth(word, getCanvasFontSize(text));
      let linkWidth = text.offsetWidth - topCatsPadding - topCatsMargin - exploreCatsParentPadding;
      let charWidth = getTextWidth("A", getCanvasFontSize(text)); //subtract a char from linkwidth for precision
      if (parseInt(textWidth, 10) >= (parseInt(linkWidth - charWidth, 10))) {
        isWrap = true;
      }
    }
    if (!isWrap) {
      text.classList.remove("lines-1");
      text.classList.add("lines-2");
    }
    if (words.length == 1 || isWrap) {
      text.classList.remove("lines-2");
      text.classList.add("lines-1");
    }
  }
};

const isSameObject = (oldArray, newArray) => {
  //Sometimes the order of properties changes so we have to sort it before comparing 
  //the objects.
  const sortedOldArray = getSortedArrayObject(oldArray);
  const sortedNewArray = getSortedArrayObject(newArray);
 
  return JSON.stringify(sortedOldArray) === JSON.stringify(sortedNewArray)
}
 
const getSortedArrayObject = (arr) => {
  const sortedArray = [];
 
  for (let obj of arr) {
    const sortedKeys = Object.keys(obj).sort();
    const sortedObj = {};
 
    for (let key of sortedKeys) {
      sortedObj[key] = obj[key];
    }
    sortedArray.push(sortedObj);
  }
  return sortedArray;
}
