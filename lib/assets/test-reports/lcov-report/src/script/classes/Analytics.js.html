//
// BN Localytics documentation:
// https://wiki.barnesandnoble.com/confluence/pages/viewpage.action?spaceKey=Product&title=Localytics+Instrumentation
//
import { attributes, timeBands } from '../lib/constants/analytics'
 
 
class Analytics {
 
  constructor() {
    this._enterTime = 0;
    this._currentScreen = 'NA';
    this._previousScreen = 'NA';
  }
 
 
  enter (screen) {
    console.log('Analytics: ENTER', screen);
    window.Device.tagScreen({
      location: screen
    });
 
    // Set state
    this._enterTime = Date.now();
    this._previousScreen = this._currentScreen;
    this._currentScreen = screen;
  }
 
 
  exit (event) {
    // Set the previous screen value
    if (event.data && event.data[attributes.previousScreen] !== undefined) {
      event.data[attributes.previousScreen] = this._previousScreen;
    }
 
    // Set the time on screen values
    let timeBand = timeBands[0];
    const timeSpent = Math.floor((Date.now() - this._enterTime) / 1000);
    for (let key in timeBands) {
      if (timeSpent > key) {
        timeBand = timeBands[key];
      }
    }
    if (event.data && event.data[attributes.timeOnScreenRaw] !== undefined) {
      event.data[attributes.timeOnScreenRaw] = timeSpent;
    }
    if (event.data && event.data[attributes.timeOnScreen] !== undefined) {
      event.data[attributes.timeOnScreen] = timeBand;
    }
 
    let data = {
      event: event.eventName,
      data: event.data,
      context: event.data // No longer used by native, keep for backward compatibility
    };
 
    // Fire the event
    console.log('Analytics: EXIT', data);
    window.Device.tagEvent(data);
  }
}
 
export default new Analytics()
