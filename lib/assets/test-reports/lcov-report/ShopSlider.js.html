class ShopSlider extends PureComponent {
 
  constructor(props) {
    super(props);
    this.slider = null; // ref in the render function
    this.isEPD = window.Device.shopPlatform === 'epd' || window.Device.shopPlatform === 'quill';
 
    this.state = {
      currentSlide: 0,
      isTab: !this.isEPD && window.innerWidth > 549,
      speed: this.isEPD ? 0 : 500
    };
 
    // Bind Methods
    this.next = this.next.bind(this);
    this.previous = this.previous.bind(this);
    this.previousState = this.previousState.bind(this);
  }
  componentWillMount() {
    this.setState({
      previousSlide: sessionStorage.getItem('f2PodPosition') || 0,
    });
  }
  componentDidMount() {
    const previousSliderPosition = sessionStorage.getItem('f2PodPosition');
    this.setState({ currentSlide: previousSliderPosition });
    sessionStorage.removeItem('shop-category-pod-clicked');
    sessionStorage.removeItem('f2PodPosition');
    this.slider.slickGoTo(previousSliderPosition, true);
  }
 
  componentWillUpdate(nextProps, nextState) {
    // EPD - don't animate slide transitions
    Eif (this.state.currentSlide !== nextState.currentSlide) {
      this.slider.slickGoTo(nextState.currentSlide, true);
    }
  }
 
  changeAnimationSpeed = (oldIndex, newIndex) => {
    let lastIndex = this.slider.props.children.length - 1;
    if (!this.isEPD) { // for EPD, animation duration was already set as 0 .
      if (window.Device.shopPlatform === platforms.android &&
        (oldIndex > newIndex || (oldIndex == 0 && newIndex == lastIndex))) {
        this.setState({ speed: 0 }) // skip animation while sliding left
      }
      else if (window.Device.shopPlatform === platforms.ios &&
        ((oldIndex == 0 && newIndex == lastIndex) ||
          (oldIndex == lastIndex && newIndex == 0))) { // for iOS skip animation in last slide
        this.setState({ speed: 0 })
      }
      else {
        this.setState({ speed: 500 }) //set animation duration
      }
    }
  }
 
  previous() {
    let prevSlide = this.state.currentSlide - 1;
    if (this.state.currentSlide === 0) {
      prevSlide = this.props.children.length - 1;
    }
    this.setState({ currentSlide: prevSlide });
  }
 
  next() {
    let nextSlide = this.state.currentSlide + 1;
    if (nextSlide === this.props.children.length) {
      nextSlide = 0;
    }
    this.setState({ currentSlide: nextSlide });
 
  }
  
  previousState(current, next) {
    this.setState({previousSlide: current, currentSlide: next });
    sessionStorage.setItem('f2PodPosition', next);
  }
 
  renderEPDArrows() {
    return (
      <div className="epd-slider-arrows">
        <div className="butt epd-button-prev" onClick={ this.previous }>
          {this.isEPD ? <img alt="left arrow" src={ ArrowRightEpd } /> : <img alt="left arrow" src={ ArrowRight } />}
        </div>
        <div className="butt epd-button-next" onClick={ this.next }>
          {this.isEPD ? <img alt="right arrow" src={ ArrowRightEpd } /> : <img alt="right arrow" src={ ArrowRight } />}
        </div>
      </div>
    );
  }
 
 
  render() {
    return (
      <section aria-roledescription="carousel">
      <span className="shop-slider">
        <Slider className="carousel"
            aria-live="off"
            lazyLoad
            ref={ el => this.slider = el }
            arrows={ false }
            swipe={ true }
            touchThreshold={ 50 }
            dots={ false }
            prevArrow={ null }
            nextArrow={ null }
            infinite={ true }
            speed={this.state.speed}
            slidesToShow={ !this.state.isTab ? 1.09 : 1.30 }
            useTransform={false}
            beforeChange= {this.changeAnimationSpeed}>
          { this.props.children }
        </Slider>
      </span>
      </section>
    );
  }
}
 
export default ShopSlider;
