            context_locale: 'en_US',
            countryCode: urlParams.countryCode,
            accountId: urlParams.accountId,
            profileId: urlParams.profileId,
            storeCode: urlParams.storeCode,
            productDeviceId: device.productDeviceId
          }, urlParams, options);
        }
        else {
          let store = this.storeCode ? this.storeCode : this.computeStoreCode(listFormatType, urlParams.storeCode);//categories
          if(store) {
            urlParams.storeCode = store;
          }
          this.loadList(urlParams, options);

    }

  //
  // Fetch the list content
  // 
  loadList(urlParams, options = {}) {
    if (options.loadMore) {
      this.setState({ isFetchingMore: true });
    }
    else {
      this.setState({ isFetchingList: true });
    }
    if (this.setRecomAfterRefresh) {
      fetchF1(urlParams, (err, resp) => {
        if (err) {
          console.error('Error fetching F1 recommendations.', err);
          resp = { eans: [], pods: [], recommendations: {} };
        }
        this.setState({ isFetchingF1: false });
        const listData = {
          data: resp.recommendations || {},
          eanData: resp.eans
        };
        this.props.setList(listData);
        this.displayF3 = true;
      });
    }
    else {
      fetchList(urlParams, (err, data) => {
        if (err) {
          console.error('Error fetching list items.', err.message);
          window.Device.showError('Error fetching channels items.' + err.message);
          data = { eans: [], listData: { eans: [] } };
        }
        else {
          if (options.loadMore) {
            this.setState({ isFetchingMore: false });
          }
          else {
            this.setState({ isFetchingList: false });
          }
        }
 
        //Handled data mismatch (pending requests overwrites the actual response) in Coming Soon and New Releases.
        if (window.location.href.match("pubdate") && data.listData.id) {
          var pubDate = this.drefFromUrl(data.listData.id, /^pubdate/)
          if (!pubDate || !window.location.href.match(pubDate)) {
            return;
          }
        }
 
        const listData = {
          eanData: data.eans,
          data: {
            total: data.listData.total,
            loadedListId: data.listId || this.drefFromUrl(data.listData.id, /^dref/), // DREF lists don't have an "id", get the dref from the search URL
            eans: data.listData.eans,
            displayAlaCartePrice: data.listData.displayAlaCartePrice,
            displayAllFormats: data.listData.displayAllFormats,
            displayCredit: data.listData.displayCredit,
            freeList: data.listData.freeList,
            comingSoon: data.listData.comingSoon,
            maxPrice: data.listData.maxPrice,
            listFormatType: data.listData.listFormatType
          }
        };
        // If we fetched search results, we already have some of the details
        if (this.props.dref) {
          this.props.setListDetails({ data: data.listData });
        }
 
        // Demand load more list items (triggered by scrolling)
        if (options.loadMore || this.isFetchingMore) { // Demand load
          this.props.updateList(listData);
        }
 
        // Initial list list load
        else {
          this.props.setList(listData);
        }
 
        this.alignTitles();
        if (!this.isEpd) {
          window.Device.modifyHeader({
            title: getTitle(this.props.isChildProfile),
            enableBack: true,
            hideTools: this.isCategories
          });
        }
        this.displayF3 = true;
 
        this.updateLibrary();
      });
    }
  }
 
 
  //
  // We need to call the listDetails API to get the total number of items in the
  // list.
  //
  // List title/description are not returned by Search API. We must get it 
  // from the listDetails API as well, if we do not have them.
  // 
  loadListDetails(userParams, urlParams, options) {
    this.setState({ isFetchingListDetails: true });
    fetchListDetails(userParams, (err, response) => {
 
      // Set not fetching so we can still demand load even if there was an 
      // error getting details
      this.setState({ isFetchingListDetails: false });
 
      if (err) {
        console.error('Error fetching list details.', err.message);
        window.Device.showError('Error fetching list details.' + err.message);
        response = {};
      }
 
      const listData = {
        listId: userParams.listId,
        data: response
      };
 
      let store = this.computeStoreCode(listData.data.listFormatType, urlParams.storeCode);
      this.storeCode = store;
 
      this.props.setListDetails(listData);
      if(store) {
        urlParams.storeCode = store;
      }
      this.loadList(urlParams, options);
 
    });
  }
 
  computeStoreCode(listFormatType, storeCode) {
    const isChildProfile = this.props.customer.childProfile;
    const isAudiobooks = window.SHOW_AUDIOBOOKS;
 
    if (this.isCategories && isAudiobooks) {
      return isChildProfile ? storeCodes[storeCodeMap.kidsdigaud].code : storeCodes[storeCodeMap.digaud].code;
    }
    //Check for Audiobooks and type mixed list
    if (isAudiobooks && (this.isCategories || listFormatType !== 'Mixed')) {
      return isChildProfile ? storeCodes[storeCodeMap.kidsdigaud].code : storeCodes[storeCodeMap.digaud].code;
    }
    // Mixed has to be handled for both ebook and audiobook shop
    return listFormatType === 'Mixed' ? (isChildProfile ? storeCodes[storeCodeMap.kidsebookda].code : storeCodes[storeCodeMap.ebookda].code) : storeCode;
  }
 
 
  drefFromUrl(url, matchRegEx) {
    if (url) {
      const paramPairs = url.split('?').pop().split('&');
      const drefPair = paramPairs.find(d => d.match(matchRegEx));
      if (drefPair)
        return decodeURIComponent(drefPair.split('=').pop());
    }
  }
 
 
  handleSort(sortKey) {
    this.offset = 1; // Reset to the top of the list when sorting.
    this.oldListLength = 0;
    this._sortUsed = true;
    if (this.isEpd && this.state.sortKey !== sortKey) {
      console.log("Sort Changed...." + sortKey);
      window.resetPageNo = true;
      if (window.paginationComponent) {
        if (this.isCategories) {
          let categoryPageNo = Math.ceil(this.props.hasTopLists / 2) + 1;
          window.categoryPageReset = true;
          window.paginationComponent.gotoPage(categoryPageNo);
        } else {
          window.paginationComponent.gotoPage(1);
        }
      }
    }
    this.setState({ sortKey });
    this.allLoaded = false; // Reset allLoaded flag to false when sorting
  }
 
  handleFilter(audioFilter) {
    this.offset = 1; // Reset to the top of the list when sorting.
    this.oldListLength = 0;
    //this._sortUsed = true;
    this.setState({ audioFilter });
  }
 
  handleGrid(sortKey, audioFilter) {
    sessionStorage.setItem("showList", sortKey);
    this.setState({ showGrid: sortKey });
  }
 
 
  demandLoad(scrollPos) {
    if (scrollPos < 0) { return; }
    this.allLoaded = this.allLoaded && !window.sortChanged;
    if (this.allLoaded || this.isLoadingMore || this.state.isFetchingList || this.state.isFetchingMore || this.props.subscription) {
      return;
    }
    window.sortChanged = false;
    this.isLoadingMore = true;
    this.offset += this.itemCount;
    this._demandLoadCount++; // Analytics value
 
    // Send demand load count to parent component (like category results) for analytics
    if (this.props.onDemandLoad) {
      this.props.onDemandLoad(this._demandLoadCount);
    }
    this.openList({ loadMore: !this.isEpd ? true :  (this.isEpd && scrollPos.type === 'onEPDLastPage') ? true : false });
  }
