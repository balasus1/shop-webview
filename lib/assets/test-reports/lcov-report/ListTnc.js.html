class ListTnc extends Component {
 
  constructor() {
    super();
 
    // Local state for this component only
    this.state = {
      showTncPopup: false
    };
 
    // Bind methods
    this.openPopUp = this.openPopUp.bind(this);
    this.closePopUp = this.closePopUp.bind(this);
  }
 
 
  render() {
 
    return (
      <div className="list-tnc">
        <div className="tnc-link" onClick={ this.openPopUp }>
          { this.props.tnc.link }
        </div>
        {
          this.state.showTncPopup &&
          <div className="tnc-popup-container">
            <div className="tnc-popup tnc-message">
              <div className="tnc-header" onClick={ this.closePopUp } />
              <div className="text" dangerouslySetInnerHTML={ {__html: this.props.tnc.message} } />
            </div> 
          </div>
        }
      </div>
    );
  }
 
  openPopUp(evt){
    if (evt.target.classList.contains('tnc-link')) {
      this.setState({showTncPopup: true});
    }
  }
 
  closePopUp(evt) {
    if (evt.target.classList.contains('tnc-header')) {
      this.setState({showTncPopup: false});
    }
  }
}
 
export default ListTnc;
