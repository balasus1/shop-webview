/**
 * This component renders Nook Channels. Nook Channels is a specially-created 
 * list of lists, built in the LMS.
 *
 * Technically, Nook Channels is a "list of lists of lists" since there are two
 * lists of lists: affinity & popular.
 * HOWEVER - we only render the popular lists and treat it like any other LoL
 * 
 * As the user scrolls to the bottom of Channels, more content is automatically
 * fetched and rendered. This is called "demand loading".
 */

class Channels extends Component {
 
  constructor() {
    super();

    this.isEpd = window.location.href.match(platforms.epd);

    this.oldPopularListsCount = 0;

  initAnalytics() {
    this.analyticsEvent = Object.assign({}, events.channelsViewed);
    this.analyticsEvent.data = Object.assign({}, events.channelsViewed.data);
  }
 
 
  componentDidMount() {
 
    // Load the content
    this.openChannels();
 
    // Setup a scrolling handler to load more list items on scroll
    this.scroller = new Scroller();
    this.scroller.register(this.demandLoad, {
      scrollOffset: this.demandLoadOffset
    });
    this.refinementType = this.props.refinementType;
    // Analytics
    analytics.enter(window.SHOW_AUDIOBOOKS ? screens.audioBooks.channels : screens.ebooks.channels);
    this.initAnalytics();

  componentWillMount() {

    // Always start at the top of page
    window.scroll(0,0);
  }
 
 
  componentWillUnmount() {
 
    // Disable the scroller so events don't fire after this component is gone
    this.scroller.unregister();
 
    // Localytics Event
    this.sendAnalytics();

  componentDidUpdate() {
    if (window.location.href.match(platforms.android) && this.refinementType !== this.props.refinementType) {
      window.location.reload(false);  
    }
 
    // If demand loading has finished, check if all lists for each spot have been loaded
    if (this.isLoadingMore && !this.state.isFetchingMoreChannels) {

  //
  // Load the initial Nook Channels data
  // 
  loadChannels(urlParams, options={}) {
    if (options.loadMore) {
      this.setState({ isFetchingMoreChannels: true });
    }
    else {
      this.setState({ isFetchingChannels: true });
    }
 
    fetchChannels(urlParams, (err, response) => {
      if (err) {
        console.error('Error fetching channels items.', err.message);
        window.Device.showError('Error fetching channels items.' + err.message);        
        response = {affinityLists:[], popularLists:[], eans: []};
      }
      else {
        this.setState({ isFetchingChannels: false, isFetchingMoreChannels: false });
      }
 
      // Demand load more lists (triggered by scrolling)
      if (options.loadMore) {
        this.props.updateChannels(response);
      }
 
      // Initial channels load
      else {
        this.props.setChannels(response);
      }
    }, {cachedResponse: false});
  }
 
 
  //
  // Callback when the user scrolls to the bottom
  // 
  demandLoad(scrollPos) {
    if (this.allLoaded || (this.isLoadingMore)) {
      return;
    }
    this.isLoadingMore = true;
    this.offset += this.listsPerSpot;
    this.openChannels({loadMore: true});
  }
