import React, { Component } from 'react'
import platforms from '../lib/constants/platforms'
import { NavLink } from 'react-router-dom'
import ShopRetry from './ShopRetry'
import moreButton from '../../images/btn_more_list_enter.png'
 
import '../../style-dist/components/F5Lol.css'
import textReplace from '../lib/i18n'
if (window.location.href.match(platforms.epd))
  require('../../style-dist/components/F5Lol.epd.css')
if (window.location.href.match(platforms.quill))
  require('../../style-dist/components/F5Lol.quill.css')
 
 
class F5Lols extends Component {
 
  constructor(props) {
    super(props);
 
    // Define empty lists to show while loading
    this.loadingLists = new Array(4).fill({ listCloudId: 1, title: '', description: '' });
 
    this.state = {
      f5s: []
    };
  }
 
 
  componentWillMount () {
    this.setData(this.props);
  }
 
 
  componentWillReceiveProps (nextProps) {
    this.setData(nextProps);
  }
 
 
  setData(props) {
    // Remove duplicates when F4 and F5 data is provided
    let f5s = [];
    if (props.data.f5.length && props.data.f4.length && !this.state.f5s.length) {
      for (let f=0; f<props.data.f5.length; f++) {
        const f5 = props.data.f5[f];
        const isDuplicate = props.data.f4.find( f4 => f4.listCloudId === f5.listCloudId);
        if (!isDuplicate) {
          f5s.push(f5);
        }
      }
      this.setState({ f5s });
    }
  }
 
 
  renderF5List(list, i) {
    const listId = list.listCloudId;
    const shopPlatform = window.Device.shopPlatform;
 
    return <NavLink className="f5-tile" key={i} 
          to={ `/${shopPlatform}/list/${listId}` }
          onClick={ (evt) => this.handleClick(evt, list) }>
      <div>
        <div className="title ellipses lines-2">{ list.title }</div>
        <div className="description ellipses lines-2">{ list.description || textReplace(KEY.recommended_for_you) }</div>
        {
          shopPlatform === "epd"  || shopPlatform === "quill" ?
            <img alt={textReplace(KEY.more)} src={ moreButton }/>
            :
            <div className="cta ellipses">{textReplace(KEY.cta_explore)}</div>
        }
      </div>
    </NavLink>
  }
 
 
  render() {
    let loading = false;
    let data = this.state.f5s;
    const shopPlatform = window.Device.shopPlatform;
    
    if (this.props.isFetchingF5) {
      loading = true;
      data = this.loadingLists;
    }
 
    const loadingAriaAttribs = {
      "aria-busy": true, "aria-live":"off"
    };
 
    // Render the retry button
    if (!this.props.isFetchingF5 && this.state.f5s && !this.state.f5s.length) {
      return <div className="f5-lists"><ShopRetry /></div>;
    }
 
    let f5s = data.slice(0, 4).map(this.renderF5List, this);
    const title = textReplace(KEY.more_lists);
    const subtitle = textReplace(KEY.see_more);
    const titleEncoded = encodeURIComponent(title);
    const listIds = data.map(f5 => f5.listCloudId);
    const lolLink = `/${shopPlatform}/listoflists/${listIds.slice(5, 11).join(',')}/${titleEncoded}`
 
    return <div {...(loading ? loadingAriaAttribs : {"aria-live":"off"})} className={ 'f5-lists' + (loading ? ' loading' : '') }>
      <NavLink role="heading" aria-level="2" className="title" to={ lolLink }>
       <span role="link"> { title }</span>
      </NavLink>
      <div className="f5-tile-container">
      { f5s }
      </div>
      <NavLink className="subtitle" to={ lolLink }>
        { subtitle }
      </NavLink>
    </div>;
  }
 
 
  handleClick(evt, list) {
 
    // Send Analytics
    this.sendData(list.title);
  }
 
 
  // Send back the list analytics
  sendData(title, eanPos=0) {
    const data = {
      listTitle: title,
      listEanPosition: eanPos,
      listPosition: this.props.index
    };
 
    if (eanPos > 0) {
      this.props.onCoverClick(data);
    }
 
    else {
      this.props.onListClick(data); // TODO analytics
    }
  }
}
 
export default F5Lols;
