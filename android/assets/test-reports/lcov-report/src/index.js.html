/**
 * This file initializes the entire Shop application, it is the first file to run.
 * It loads the React component called "Root" which defines our application structure.
 */
 
// eslint-disable-next-line
import './script/lib/polyfill'
 
import React from 'react'
import ReactDOM from 'react-dom'
import { Provider } from 'react-redux'
import { createStore } from 'redux'
import browserReady from './script/lib/native/browserReady'
import deviceTest from './script/lib/native/deviceTest'
import shopReducer from './script/reducers/index'
import Root from './script/components/Root'
 
import './style-dist/index.css'
 
const AKAMAI_HOST = "docs.nook.com";
//
// Control a global "debug" flag that shows "HELPFUL" info in the UI
//
window.NOOK_DEBUG = false;
window.NOOK_DEV = false;
window.LOCALHOST = !!window.location.href.match("localhost") || 
!!window.location.href.match(AKAMAI_HOST);
 
 
//
// Create the Redux store, this is where all of our application state is stored.
// 
// The reducer is a special function that listend for Redux "dispatch" events and
// returns a new application state.
// 
const store = createStore(shopReducer, window.__REDUX_DEVTOOLS_EXTENSION__ && window.__REDUX_DEVTOOLS_EXTENSION__());
 
 
browserReady(() => {
  const root = document.getElementById('root');
  ReactDOM.render(
    <Provider store={store}>
      <Root />
    </ Provider>,
    root
  );
});
 
 
window.getHtmlShopVersion = () => {
  return process.env.REACT_APP_VERSION;
}
 
 
window.deviceTest = deviceTest;
 
 
//
// Keep track of the previous page the user visited
//
window.onhashchange = event => {
  sessionStorage.oldURL = event.oldURL;
  sessionStorage.newURL = event.newURL;
}
 
 
window.onerror = function(m, s, l, c, err) {
  console.log("==> Uncaught Exception");
  console.log("  > " + m);
  console.log("  > file: " + s);
  console.log("  > line:" + l);
  console.log("  > col :" + c);
  console.log("  > err :" + err);
};
