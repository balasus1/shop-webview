import actions from '../lib/constants/actionTypes';
 
const initialState = {
  affinityLists: [],
  popularLists: []
};
 
const channelsView = (state = initialState, action) => {
  let stateCopy = Object.assign({}, state);
 
  const removeDuplicates = (lists) => {
    let temp = [];
    for (let l = 0; l < lists.length; l++) {
      let list = lists[l];
      const isDuplicate = temp.find(t => t.title === list.title);
      if (!isDuplicate) {
        temp.push(list);
      }
    }
    return temp;
  }
 
  switch (action.type) {
    
    case actions.SET_CHANNELS_ITEMS:
      return Object.assign(stateCopy, {
        affinityLists: action.affinityLists,
        popularLists: action.popularLists
      });
    
    case actions.ADD_CHANNELS_ITEMS:
      let existingAffinity = stateCopy.affinityLists;
      let existingPopular = stateCopy.popularLists;
      return Object.assign(stateCopy, {
        affinityLists: removeDuplicates(existingAffinity.concat(action.affinityLists)),
        popularLists: removeDuplicates(existingPopular.concat(action.popularLists))
      });
    
    default:
      return state;
  }
}
 
export default channelsView
