import actions from '../lib/constants/actionTypes';
 
const eanData = (state = {}, action) => {
 
  switch (action.type) {
    case actions.ADD_EAN_DATA:
 
      // Add the existing eanData to the new eanData
      return Object.assign(action.eans, state);
    
    default:
      return state;
  }
}
 
export default eanData
