import actions from '../lib/constants/actionTypes'
 
const initialState = {
  f1: {},
  f2: {
    data: [],
    status: true
  },
  favCategories: [],
  f3: {
    links: [],
    affinities: {}, // Some links have associated affinities. Map them here for easy lookup
    status: true
  },
  f4: {
    data: [],
    eans:{},
    status: true
  },
  f5: []
};
 
const forMeData = (state = initialState, action) => {
  let stateCopy = Object.assign({}, state);
 
  switch (action.type) {
 
    case actions.SET_F1_DATA:
      stateCopy.f1 = action.f1Data;
      return stateCopy;
    
    case actions.SET_F2_DATA:
      const f2Copy = Object.assign({}, stateCopy.f2)
      f2Copy.data = action.f2Data;
      f2Copy.status = action.status;
      stateCopy.f2 = f2Copy;
      return stateCopy;
    
    case actions.SET_FAV_CATEGORIES_DATA:
      stateCopy.favCategories = action.favCategoriesData;
      return stateCopy;
 
    case actions.UPDATE_FAV_CATEGORIES_DATA:
        stateCopy.favCategories = action.UPDATE_FAV_CATEGORIES_DATA;
        return stateCopy;
 
    case actions.SET_F3_DATA:
      const f3Copy = Object.assign({}, stateCopy.f3);
      f3Copy.links = action.f3Data.links;
      f3Copy.status = action.status;
      stateCopy.f3 = f3Copy;
 
      const affinityMap = {};
      action.f3Data.links.forEach( (link, idx) => {
        let affinityIds = link.affinity_ids || [];
        affinityIds.forEach( affinity => {
          affinityMap[affinity] = idx; // Store the index of the link in the links array
        });
      });
      stateCopy.f3.affinities = affinityMap;
      return stateCopy;
    
    case actions.SET_F4_DATA:
      const f4Copy = Object.assign({}, stateCopy.f4)
      f4Copy.data = action.f4Data;
      f4Copy.status = action.status;
      stateCopy.f4 = f4Copy;
      return stateCopy;
 
    case actions.SET_F5_DATA:
      stateCopy.f5 = action.f5Data;
      return stateCopy;
    
    default:
      return state;
  }}
 
export default forMeData
