import React, { Component } from 'react'
import ebookLCD from '../../images/ebook-icon/ebook_lcd.svg'
import audioBookLCD from '../../images/audiobook-icon/audiobook_lcd.svg'
import ebook from '../../images/ebook-icon/ebook.svg';
import audioBook from '../../images/audiobook-icon/audiobook.svg';
import platforms from '../lib/constants/platforms'
import textReplace from '../lib/i18n';
import KEY from '../lib/constants/languages/keys';
 
class ListCoverIcons extends Component {
 
  render() {
    let showEbook = false;
    let showAudioBook = false;
    let bothFormats = false;
    let displayCredit = this.props.displayCredit && this.props.availableCredit !== undefined;
    let showBothPrices = this.props.displayAlaCartePrice && displayCredit;
    let displayAllFormats = this.props.displayAllFormats;
    let displayAlaCartePrice = this.props.displayAlaCartePrice;
    let availableCredit = this.props.availableCredit;
    this.isEPD = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
 
    if (window.location.hash.match("categories") || window.location.hash.match("search")) { // Handling for all search related lists.
      displayAllFormats = true; // Though we set default true, icons will be handled based on this.props.availableFormat flag
      displayAlaCartePrice = this.props.priceValue !== undefined && this.props.priceValue !== '';
      displayCredit = this.props.availableCredit !== undefined;
      showBothPrices = displayAlaCartePrice && displayCredit;
    }
 
    if (displayAllFormats) { //For mixed formats
      bothFormats = (this.props.availableFormat === 'FA,ER' || this.props.availableFormat === 'ER,FA');
      showAudioBook = this.props.availableFormat === 'FA' || bothFormats;
      showEbook = this.props.availableFormat === 'ER' || bothFormats;
    }
 
    else {// For single formats
      showAudioBook = this.props.isAudio;
      showEbook = !this.props.isAudio;
    };
 
    displayCredit = displayCredit && !bothFormats && showAudioBook;
    displayAlaCartePrice = displayAlaCartePrice && (!bothFormats || window.noAudiobookSupport);
 
    return <div className="bookWraper">
      <div>
        {showEbook && <img alt={textReplace(KEY.book_icon)} className="bookImg" src={this.isEPD ? ebook: ebookLCD}></img>}
        {showAudioBook && !window.noAudiobookSupport && <img alt={textReplace(KEY.audio_book_icon)} className="bookImg" src={this.isEPD ? audioBook : audioBookLCD}></img>}
        <div className="bookPrice">
          {!window.purchaseDisabled &&
          ((!displayAlaCartePrice && !displayCredit && !bothFormats) ||
              displayAlaCartePrice || this.props.priceValue === '0.00') ? (
              this.props.price
          ) : (
              !displayCredit && <div>&nbsp;</div>
          )}
        {!window.purchaseDisabled && ((displayCredit && this.props.priceValue !== '0.00')) ? ((showBothPrices ? ` ${textReplace(KEY.or)} ` : "") + availableCredit + ` ${textReplace(KEY.credit)}`) : ""}
        </div>
      </div>
    </div>
  }
}
export default ListCoverIcons;
