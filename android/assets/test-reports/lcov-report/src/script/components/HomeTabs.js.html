class HomeTabs extends Component {
 
  constructor() {
    super();
 
    const platform = window.Device.shopPlatform;
    const basePath = `/${platform}/home/`;
    this.tabs = [
      {
        name: 'For Me',
        link: basePath + viewTypes.home.FOR_ME
      },
      {
        name: 'Categories',
        link: basePath + viewTypes.home.CATEGORIES
      }
    ]
  }
 
 
  render() {
    return (
      <div className="shop-home-tabs">
        { this.tabs.map( (tab, idx) => {
          return (
            <NavLink className="tab" activeClassName="active" to={ tab.link } key={ idx }>
              { tab.name }
            </NavLink>
          );
        }) }
      </div>
    );
  }
}
 
export default HomeTabs;
