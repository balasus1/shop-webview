import React from 'react'
import LongPress from 'react-long';
import ItemCover from './ItemCover'
import RatingStars from './RatingStars'
import ReadInStore from './ReadInStore'
import defaultBook from '../../images/default_book.png'
import ListCoverIcons from './ListCoverIcons'
 
import '../../style-dist/components/ListGridCover.css'
import '../../style-dist/components/ReadInStore.css'
import textReplace from '../lib/i18n';
import KEY from '../lib/constants/languages/keys';
 
//
// Extend the ItemCover component to allow for easy default image loading
// 
class ListGridCover extends ItemCover {
 
  constructor() {
    super();
 
    this.LONG_PRESS_TIMEOUT = 600;
 
    // Bind methods
    this.handleLongPress = this.handleLongPress.bind(this);
    this.openPdp = this.openPdp.bind(this);
  }
 
 
  render() {
    const loading = this.props.loading;
    const eanObj = this.props.eanObj;
    const isComingSoon = eanObj.isComingSoon;
    const isInLibrary = this.props.isInLibrary;
    const availableFormat = eanObj.availableFormats;
    const availableCredit = eanObj.numberOfCredits;
    const productData = this.props.productData;
    const displayAllFormats = productData ? productData.listFormatType === 'Mixed' : false;
    const displayAlaCartePrice = productData ? productData.displayAlaCartePrice === 'true' : false;
    const displayCredit = productData ? productData.displayCredit === 'true' : false;
    const isAudio = eanObj.productType === 'AUDIOBOOK';
    const narrator = eanObj.narrator ? (`${textReplace(KEY.read_by)}: ` + eanObj.narrator) : '';
    const loadingAriaAttribs = {
      "aria-busy": true, "aria-live": "off"
    };
 
    let isPriceNaN = isNaN(Number(eanObj.price))
    let priceValue = isPriceNaN ? '' : Number(eanObj.price).toFixed(2);
    let price = '';
    let freeBook = false;
 
    if (priceValue === "0.00" || priceValue === 0 || priceValue === '0') {
      price = textReplace(KEY.free);
      freeBook = true;
    }
    else if (priceValue !== '') {
      price = '$' + priceValue;
    }
 
    // If the ean object is missing or malformed show something nice
    // However, if we are still loading conten do nt show this.
    if ((!eanObj || !eanObj.ean) && !loading) {
 
      // Can show "filler" covers at the end of lists to keep the "grid" shape
      // and prevent the last cover(s) from centering
      return <div className={'list-grid-cover error not-found' + (this.props.filler ? ' filler' : '')}>
        <img className="cover" src={defaultBook} alt={textReplace(KEY.loading)} />
        <div className={window.SHOW_AUDIOBOOKS ? "audiobook-item-details" : "item-details"}>
          <RatingStars rating={0} />
          <div className="price">
          </div>
        </div>
      </div>;
    }
 
    return (
      <LongPress
      onLongPress={this.handleLongPress}
      time={this.LONG_PRESS_TIMEOUT}>
      <div {...(loading ? loadingAriaAttribs : { "aria-live": "off" })} className={'list-grid-cover' + (loading ? ' loading' : '')}
        ref={el => this.coverEl = el}
        onClick={this.openPdp}>
          {
            !this.state.isImageEmpty ?
              <div><img className="cover" src={this.state.imageSrc} alt={eanObj.title} />
                {(isInLibrary || this.state.purchase) &&
                  <div className={"purchased" + (isComingSoon ? " pre-ordered" : "")} > <span>{isComingSoon ? textReplace(KEY.pre_ordered) : textReplace(KEY.purchased)}</span></div>
                }
              </div>
              :
              <div className="no-image-cover">
                <div className="title ellipses lines-2">
                  {eanObj.title}
                </div>
                <div className="author ellipses lines-2">
                  {eanObj.author}
                </div>
              </div>
          }
 
        <div className="readinstore-container">
          {!isComingSoon && !isInLibrary && !freeBook && <ReadInStore inStore={this.props.inStore && this.props.eanObj.isBrowsable === true && this.props.eanObj.browseMinutes > 0} />}
        </div>
 
        {
          // Only show the cover on EPD Home
          window.Device.isEpdHome ? '' :
 
            <div className="item-details">
              {
                this.props.showDetails ? <div className="title-author">
                  <div className="title ellipses lines-2">
                    {eanObj.title}
                  </div>
                  { !loading && <div className="author ellipses">By {eanObj.author}</div>}
                  <div>
                    <div className="narrator ellipses">
                      {narrator}
                    </div>
                  </div>
                </div> : ''
              }
 
              {isComingSoon ?
                <div className="pre-order">{textReplace(KEY.pre_order)}</div>
                :
                <RatingStars rating={eanObj.rating} />
              }
 
              <div className="price">
              { !isInLibrary && !isPriceNaN ?
                    <ListCoverIcons
                      isAudio={isAudio}
                      isGrid={true}
                      priceValue={priceValue}
                      price={price}
                      availableFormat={availableFormat}
                      availableCredit={availableCredit}
                      displayAllFormats={displayAllFormats}
                      displayAlaCartePrice={displayAlaCartePrice}
                      displayCredit={displayCredit}>
                    </ListCoverIcons> : '\u00A0'}
              </div>
            </div>
        }
      </div>
      </LongPress>
 
    );
  }
 
  setPurchaseFlag = () => {
    this.setState({ purchase: true });
    this.props.updateLibrary();
  }
 
  openPdp(evt) {
    evt.preventDefault();
    const eanObj = this.props.eanObj;
    this.props.onClick(this.props.index); // Send back the EAN position in the list
 
    // Check for an Internet connection (only EPD Home shows offline - native caches it, not the normal Shop)
    window.Device.isConnected(connected => {
      if (connected) {
        window.sessionStorage.setItem("hash_before_entering_pdp",window.location.hash);
        window.listCover = this;
        window.Device.showProductDetails(eanObj.title, eanObj.ean);
      }
      else {
        window.Device.showToast({
          message: textReplace(KEY.internet_connection_needed)
        });
      }
    });
  }
 
 
  handleLongPress(evt) {
    const eanObj = this.props.eanObj;
    const imageUrl = this.state.imageSrc;
 
    const rect = this.coverEl.getBoundingClientRect();
    const centerY = Math.round(rect.top + (rect.height / 2));
    const centerX = Math.round(rect.left + (rect.width / 2));
 
    console.log('LONG PRESS');
    window.sessionStorage.setItem("hash_before_entering_pdp",window.location.hash);
    window.listCover = this;
    window.Device.showManageActionSheet(eanObj.ean, centerX, centerY, imageUrl);
  }
}
 
export default ListGridCover;
