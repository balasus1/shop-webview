import { connect } from 'react-redux'
import actionTypes from '../lib/constants/actionTypes'
import { setAppActive, setIsInStore } from '../lib/actions'
import Shop from '../components/Shop'
 
const mapStateToProps = (state, ownProps) => {
  return {
    configData: state.configData,
    sessionData: state.sessionData,
    affinityData: state.affinityData
  }
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
  return {
 
    setFetchingAffinity: (isFetching) => {
    },
 
    // Load the session data about the device and the user from the native code
    setSession: (sessionData) => {
      dispatch({
        type: actionTypes.SET_APP_SESSION,
        sessionData
      });
    },
 
    // UPdate the Shop config, and also the session data so it looks like the device ID came from native
    setConfig: (sessionData, config) => {
      dispatch({
        type: actionTypes.SET_CONFIG,
        config,
      });
 
      sessionData.device.productDeviceId = config.productDeviceId;
      dispatch({
        type: actionTypes.SET_APP_SESSION,
        sessionData,
      })
    },
 
    // Load user affinities from the Recommender via the Assembler
    setAffinities: (affinities) => {
      dispatch({
        type: actionTypes.SET_AFFINITY_DATA,
        affinities
      });
    },
 
    // Boolean used to determine if the app is in the foreground
    toggleAppActive: (isAppActive) => {
      dispatch(setAppActive(isAppActive));
    },
 
    // Boolean used to determine if user is in a B&N store
    toggleInStore: (inStore) => {
      dispatch(setIsInStore(inStore));
    }
  }
}
 
const LoadShop = connect(
  mapStateToProps,
  mapDispatchToProps
)(Shop)
 
export default LoadShop
