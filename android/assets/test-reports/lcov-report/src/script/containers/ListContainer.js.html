import { connect } from 'react-redux'
import actionTypes from '../lib/constants/actionTypes'
import List from '../components/List'
 
const mapStateToProps = (state, ownProps) => {
  return {
    customer: state.sessionData.customer,
    device: state.sessionData.device,
    eanData: state.eanData,
    data: state.listData
  }
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
 
  const updateEanData = (eanData) => {
    dispatch({
      type: actionTypes.ADD_EAN_DATA,
      eans: eanData
    });
  };
 
  return {
    
    // Set initial list items
    setList: (listData) => {
      updateEanData(listData.eanData)
      dispatch({
        type: actionTypes.SET_LIST_ITEMS,
        listData: listData.data
      });
    },
 
    // Append additional list items to the existing data
    updateList: (listData) => {
      updateEanData(listData.eanData)
      dispatch({
        type: actionTypes.ADD_LIST_ITEMS,
        listData: listData.data
      });
    },
 
    // Set the list details (ID, title, description)
    // TODO: Use this to get total after filtering the listData
    setListDetails: (listData) => {
      dispatch({
        type: actionTypes.SET_LIST_DETAILS,
        listId: listData.listId,
        listTotal: listData.data.total,
        listTitle: listData.data.title,
        listDescription: listData.data.description
      });
    }
  }
}
 
const ListContainer = connect(
  mapStateToProps,
  mapDispatchToProps
)(List)
 
export default ListContainer
