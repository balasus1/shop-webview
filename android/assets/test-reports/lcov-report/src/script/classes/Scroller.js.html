//
// Register handlers to be called on scroll actions.
// The registered callback should call the provided callback when complete.
// 
class Scroller {
 
  register(callback, options={}) {
    this.callback = callback;
    this.scrollPercent = options.scrollPercent;
    this.scrollOffset = options.scrollOffset;
    this.scrollGreaterThan = options.scrollGreaterThan;
    this.scrollLessThan = options.scrollLessThan;
    this.scrollCallback = this._calcPosition.bind(this);
 
    window.addEventListener('scroll', this.scrollCallback, true);
  }
 
 
  unregister() {
    window.removeEventListener('scroll', this.scrollCallback, true);
  }
 
 
  _calcPosition(event) {
    let trigger;
    let pos;
 
    const target = event.target;
    const isDocument = target === document;
    const scrollPos = isDocument ? window.scrollY : target.scrollTop;
    const scrollingHeight = isDocument ? document.body.scrollHeight : target.scrollHeight;
 
    if (this.scrollPercent) {
      trigger = this.scrollPercent/100;
      pos = (scrollPos + window.innerHeight) / this._getScrollHeight();
    }
    else if (this.scrollOffset) {
      trigger = this.scrollOffset;
      pos = scrollingHeight - (scrollPos + window.innerHeight);
    }
 
    if (pos <= trigger) {
      this.isCalling = true;
      this.callback(pos);
    }
 
    // Scroll distance from the top greater than (independent of the other types)
    if (this.scrollGreaterThan && scrollPos > this.scrollGreaterThan) {
      this.callback(scrollPos);
    }
 
    // Scroll distance from the top less than (independent of the other types)
    if (this.scrollLessThan && scrollPos < this.scrollLessThan) {
      this.callback(scrollPos);
    }
  }
 
 
  _getScrollHeight() {
    return Math.max(document.body.scrollHeight, document.body.offsetHeight);
  }
}
 
export default Scroller
