  defaultLocale(country) {
    switch (country) {
      case "US":
        return "en_US";
      case "GB":
        return "en_GB";
      default:
        return "en_US";
    }
  }
 
 
  determineRetailer(country) {
    return country === "US" ? "BN2" : "NOK";
  }
 
 
  determineMerchandisingLocale(country, locale) {
    // locale is not used for US/GB but is needed
    // for multi-lingual countries like CA, CH, etc
    return this.defaultLocale(country);
  }
 
 
  normalizeLocale(locale, separator = "_") {
    let parts = locale.split(/[_-]/);
    if (parts[0] !== 'en') {
      parts[1] = parts[0].toUpperCase();
    }   
    parts.join(separator);
  }
}
 
export default new locale();
