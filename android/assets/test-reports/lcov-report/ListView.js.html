import React, { Component } from 'react';
import ListContainer from '../containers/ListContainer'
 
class ListView extends Component {
 
  constructor(props) {
    super(props);
 
    this.state = {
      show: false // Initially hidden, animate in
    };
    this.listId = this.props.match.params.listId;
    this.searchParams = this.props.match.params.searchParams;
  }
 
  componentDidUpdate() {
    if ((this.listId && this.listId !== this.props.match.params.listId) || (this.searchParams && this.props.match.params.listId)) {
      window.location.reload(false);  
    }
  }
 
 
  componentDidMount() {
    // Start the component animating in after mounting completes
    window.setTimeout(() => this.setState({ show: true }));
  }
 
 
  render() {
    // Fade animation
    const opactiy = this.state.show ? '1' : '0';
    const style = {
      transition: 'opacity 300ms',
      opacity: `${opactiy}`
    }; 
    const params = this.props.match.params;
 
    // Show/hide the sort menu based on the URL
    const showSort = window.location.href.match(/\/sort/) != null;
    const showAudFilter = window.location.href.match(/\/audfilter/) != null;
    console.log('storeCode------listview',params)
 
    return <section className="shop-view list-view" style={ style } >{
      params.searchParams ?
        // Use the search API (dref, pubdate, etc.)
        <ListContainer showSort={showSort} searchParams={params.searchParams} title={params.listName || ''} showAudFilter={showAudFilter} />
      :
        // Use search to fetch a list by ID
        <ListContainer  showF3={true} showSort={ showSort } listId={ params.listId } showAudFilter={ showAudFilter } />
    }</section>;
  }
}
 
export default ListView;
