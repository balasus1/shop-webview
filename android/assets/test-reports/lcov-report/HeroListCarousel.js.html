import ListCarousel from './ListCarousel'
import ListCoverContainer from '../containers/ListCoverContainer'
import '../../style-dist/components/HeroListCarousel.css'

let isEpd = window.location.href.match(platforms.epd) || window.location.href.match(platforms.quill);
 
class HeroListCarousel extends ListCarousel {
 
  constructor(props) {
    super(props);
    this.listCarousel = this.props.listCarousel
    this.TOTAL_COVERS = window.innerWidth > 500 ? 9 : 7;
    this.initialWindowHeight = window.innerHeight;
    this.initialWindowWidth = window.innerWidth;
    this.isSquare = true;
    this.heroCarouselRef = React.createRef();
    this.MAX_DISPLAY_HEIGHT = 1000;
  }
 
  renderCovers(ean, i) {
    const eanObj = this.props.eanData[ean] || {};
    const title = this.props.list.title;
    const isInLibrary = this.state.inLibrary[ean] !== undefined &&
      (!this.state.inLibrary[ean].isSample || this.state.inLibrary[ean].isSample === 'false');
 
    let img = new Image();
    img.onload = () => {
      if (this.isSquare && !isEpd) {
        const heroCarouselNode = this.heroCarouselRef.current
        if (heroCarouselNode) {
          this.isSquare = Math.abs(img.height - img.width) < 20;
          this.isSquare ? heroCarouselNode.classList.add("square") :
           heroCarouselNode.classList.remove("square"); //square is set default in markup for audiobooks
           //but if there is any covers without square is rendered, square class will be removed here.
        }
      }
    }
    img.src = eanObj.images.ProductPageBookCover.path;
 
 
    return <ListCoverContainer key={i}
      listCarouselName={this.listCarousel}
      index={i + 1}
      eanObj={eanObj}
      isInLibrary={isInLibrary}
      onClick={idx => this.sendData(title, idx)} />;
  }
 
  componentDidMount() {
    //For You/Categories tabs are made hidden/visible on scroll. This has impact on F4 lists as
    //the cover width and height of F4 lists are calculated based on window height and width. Added this to handle difference in
    //window height. (SHOP-399).
    let platform = window.Device.shopPlatform.toLowerCase();
    if (platform === 'ios' || platform === 'android') {
      window.addEventListener('resize', this.handleResize)
      this.handleResize();
    }
  }
 
  handleResize = () => {
    let elts = document.getElementsByClassName("hero-list-carousel");
    if (elts) {
      for (let index = 0; index < elts.length; index++) {
        const element = elts[index];
        let reqHeight = this.initialWindowHeight > this.initialWindowWidth ? this.initialWindowHeight : this.initialWindowWidth;
        if (reqHeight)
          element.style.setProperty("--window-height", (reqHeight > this.MAX_DISPLAY_HEIGHT ? this.MAX_DISPLAY_HEIGHT : reqHeight) + "px");
      }
    }
  };
 
  render() {
    const list = this.props.list
    let eans = filterList(list, this.props.eanData)[0];
    this.TOTAL_COVERS = this.props.list.type == constants.TWO_ROWS ?
      (window.innerWidth > 500 ? 12 : 10) : this.TOTAL_COVERS;
 
    eans = eans.slice(0, this.TOTAL_COVERS);
    const title = list.title;
    const listId = list.listCloudId;
    const shopPlatform = window.Device.shopPlatform;
    let linkPath = `/${shopPlatform}/list/${listId}`;
    if (this.props.useSearch) {
      linkPath = `/${shopPlatform}/search/${title}/${this.props.search}`;
    }
 
    // The entire F4 is a clickable link to open the list
    return <div ref={this.heroCarouselRef} className={'hero-list-carousel' + (this.props.loading ? ' loading' : '') + 
    (this.props.list.type == constants.TWO_ROWS ? " two-rows": "") + 
    (window.SHOW_AUDIOBOOKS ? " square" : "") }>
 
      <Link role="heading" aria-level="1"
        to={linkPath}
        onClick={(evt) => this.handleClick(evt, list)}>
        <div className="title" role='link'>
          <span className={(this.isEpd ? "ellipses" : "")}>
            {title}
          </span>
          <div className="see-all" onClick={ (evt) => this.handleClick(evt, list) }>
            { constants.SEE_ALL }
            </div>
        </div>
      </Link>
 
      <div className="list">
        <div className="covers"
          ref={elem => this.listElem = elem}
          onTouchStart={this.touchStart}
          onTouchCancel={this.touchEnd}
          onTouchEnd={this.touchEnd}>
          {eans.map(this.renderCovers, this)}
        </div>
      </div>
 
      {
        // Forward arrow for EPD
        this.isEpd && <button className="arrow right" disabled={this.state.scrollPos >= (this.listElem.scrollWidth - epdConstants.DEVICE_WIDTH)} onClick={() => this.scroll(1)}>&rsaquo;</button>
      }
    </div>;
  }
 
  handleClick(evt, list) {
 
    // Send Analytics
    this.sendData(list.title);
 
    // Special handling for F1 list (nota real list)
    if (list.listCloudId === 'rec_from_ract_pod') {
      for (let index = 0; index < list.eans.length; index++) {
        if (window.Device.itemIsInLibrary(list.eans[index])) {
          list.eans.pop(index);
        }
      }
      this.props.buildF1List({
        listId: list.listCloudId,
        eans: list.eans,
        listTitle: list.title,
        listDescription: list.description || ''
      });
    }
  }
 
 
  // Send back the list analytics
  sendData(title, eanPos = 0) {
    const data = {
      listTitle: title,
      listEanPosition: eanPos,
      listPosition: this.props.index
    };
 
    if (eanPos > 0) {
      this.props.onCoverClick(data);
    }
 
    else {
      this.props.onListClick(data); // TODO analytics
    }
  }
 
 
  //
  // EPD swiping functions
  //
  swiped(e) {
    if (e.timeStamp - this.epdSwipe.timeStamp < 3000) {
      const deltaY = e.changedTouches[0].pageY - this.epdSwipe.y;
      const deltaX = e.changedTouches[0].pageX - this.epdSwipe.x;
      // Neglect swpies which are under 45 degree angle and 100 length
      try {
        if (Math.abs(deltaX) >= 10 && Math.abs(deltaX / deltaY) > 1) {
          return -1 * (deltaX / Math.abs(deltaX));
        }
      } catch (e) {
        // do nothing
      }
    }
    return 0;
  }
 
 
  touchStart(e) {
    if (!this.isEpd || this.props.noEPDSwipe) {
      return;
    }
 
    this.epdSwipe = {
      timeStamp: e.timeStamp,
      y: e.changedTouches[0].pageY,
      x: e.changedTouches[0].pageX
    }
  }
 
 
  touchEnd(e) {
    if (!this.isEpd || this.props.noEPDSwipe) {
      return;
    }
    const direction = this.swiped(e);
    this.scroll(direction);
  }
 
 
  scroll(direction) {
    const scrollPos = this.listElem.scrollLeft;
    console.log('carousel scrolling', this.listElem, scrollPos, (direction * this.EPD_SCROLL_WIDTH))
    this.listElem.scrollLeft = scrollPos + (direction * this.EPD_SCROLL_WIDTH);
 
    // This is only needed to cause a re-render on EPD when the arrows are clicked
    this.setState({ scrollPos: this.listElem.scrollLeft });
  }
 
}
 
export default HeroListCarousel;
