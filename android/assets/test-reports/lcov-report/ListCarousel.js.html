class ListCarousel extends Component {
 
  constructor(props) {
    super(props);
    // Allow EPD users to "swipe" the carousel
    this.isEpd = window.Device.shopPlatform === 'epd' || window.Device.shopPlatform === 'quill';
    this.epdSwipe = {};
    this.listElem = {}; // Populated when the list div mounts by the "ref" attribute
    this.EPD_SCROLL_WIDTH = window.Device.shopPlatform === 'epd' ? ( 4 * epdConstants.CAROUSEL_COVER_WIDTH) : (4 * epdConstants.QUILL_CAROUSEL_COVER_WIDTH );
    this.TOTAL_COVERS = window.innerWidth > 500 ? 9 : 7; // Show fewer covers on phones
    this.isCategoryPage = props.isCategoryPage;
    this.state = {
      inLibrary: {}, // EANs the user already owns (from JSNI). So we can show the correct label
      scrollPos: 0, // This is only needed to cause a re-render on EPD when the arrows are clicked
    };
 
    // Find out which EANs the user already owns so we can show the appropriate label
    this.props.list.eans = this.props.list.eanList ? 
    this.props.list.eanList : this.props.list.eans//Remove this once eanList is changed to eans in new api itself
 
    if (this.props.list.eans.length && this.props.list.eans[0] !== null) {
      window.Device.getLibraryItems(
          this.props.list.eans,
          inLibrary => {
            this.setState({ inLibrary });
          },
          err => console.error('Error getting user\'s library items.', err)
      );
    }
 
    // Bind methods
    this.swiped = this.swiped.bind(this);
    this.touchStart = this.touchStart.bind(this);
    this.touchEnd = this.touchEnd.bind(this);
  }
 
 
  componentWillUpdate (nextProps, nextState) {
    // Find out which EANs the user already owns so we can show the appropriate label
    if (nextProps.list.eans.length && nextProps.list.eans[0] && this.props.list.eans[0] === null) {
      console.log('Fetching library items');
      window.Device.getLibraryItems(this.props.list.eans, 
        inLibrary => this.setState({ inLibrary }),
        err => console.error('Error getting user\'s library items.', err)
      );
    }
  }
  
 
  renderCovers(ean, i) {
    const eanObj = this.props.eanData[ean] || {};
    const title = this.props.list.title;
    const isInLibrary = this.state.inLibrary[ean] !== undefined && 
      (!this.state.inLibrary[ean].isSample || this.state.inLibrary[ean].isSample === 'false');
 
    if (!eanObj.ean || (this.props.isRecommendation && isInLibrary)) { return false; }
 
    return <ListCoverContainer key={ i }
      productData={this.props.list}
      index={ i+1 }
      eanObj={ eanObj }
      isInLibrary={ isInLibrary }
      onClick={ idx => this.sendData(title, idx) } />;
  }
 
 
  renderDescription(description) {
    return <React.Fragment>
            { description && <div className="description">
                    <span className={(this.isEpd ? "ellipses lines-2" : "")}>
                    { description.charAt(0).toUpperCase() + description.slice(1) }
                    </span>
                  </div>}
          </React.Fragment>
  }
 
 
  render() {
    const list = this.props.list
    const isInEPDLoLPage = this.isEpd && (window.location.href.match('/channels/') || window.location.href.match('listoflists')) ;
    let eans = filterList(list, this.props.eanData)[0];
    eans = eans.slice(0, this.TOTAL_COVERS);
    const title = list.title || "_";
    const description = list.description || '';
    const listId = list.listCloudId;
    const shopPlatform = window.Device.shopPlatform;
    let list_label = title + " item";
    let linkPath = `/${shopPlatform}/list/${listId}`;
    Iif (this.props.useSearch) {

      <Link role="heading" aria-level="1"
        to={linkPath}

        <div className='title' role='link'>

            {constants.SEE_ALL}
          </div>}
        </div>
        {this.props.showDescription && this.renderDescription(description)}

      }
    </div>;
  }
 
 
  handleClick(evt, list) {
 
    // Send Analytics
    this.sendData(list.title);
 
    // Special handling for F1 list (nota real list)
    if (list.listCloudId === 'rec_from_ract_pod') {
      for (let index = 0; index < list.eans.length; index++) {
        if (window.Device.itemIsInLibrary(list.eans[index])) {
          list.eans.pop(index);
        }
      }
      this.props.buildF1List({
        listId: list.listCloudId,
        eans: list.eans,
        listTitle: list.title,
        listDescription: list.description || ''
      });
    }
  }
 
 
  // Send back the list analytics
  sendData(title, eanPos=0) {
    const data = {
      listTitle: title,
      listEanPosition: eanPos,
      listPosition: this.props.index
    };
 
    if (eanPos > 0) {
      this.props.onCoverClick(data);
    }
 
    else {
      this.props.onListClick(data); // TODO analytics
    }
  }
 
 
  //
  // EPD swiping functions

  swiped(e) {
    if (e.timeStamp - this.epdSwipe.timeStamp < 3000) {
      const deltaY = e.changedTouches[0].pageY - this.epdSwipe.y;
      const deltaX = e.changedTouches[0].pageX - this.epdSwipe.x;
      // Neglect swpies which are under 45 degree angle and 100 length
      try {
        if (Math.abs(deltaX) >= 10 && Math.abs(deltaX/deltaY) > 1) {
          return -1 * (deltaX / Math.abs(deltaX));
        }
      } catch (e) {
        // do nothing
      }
    }
    return 0;
  }
 
 
  touchStart(e) {
    if (!this.isEpd || this.props.noEPDSwipe) {
      return;
    }
    
    this.epdSwipe = {
      timeStamp: e.timeStamp,
      y: e.changedTouches[0].pageY,
      x: e.changedTouches[0].pageX
    }
  }
 
 
  touchEnd(e) {
    if (!this.isEpd || this.props.noEPDSwipe) {
      return;
    }
    const direction = this.swiped(e);
    this.scroll(direction);
  }
 
 
  scroll(direction) {
    const scrollPos = this.listElem.scrollLeft;
    console.log('carousel scrolling', this.listElem, scrollPos, (direction * this.EPD_SCROLL_WIDTH))
    this.listElem.scrollLeft = scrollPos + (direction * this.EPD_SCROLL_WIDTH);
    
    // This is only needed to cause a re-render on EPD when the arrows are clicked
    this.setState({ scrollPos: this.listElem.scrollLeft });
  }
}
