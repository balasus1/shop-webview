class EpdHome  extends Component {
 
  constructor(props) {
    super(props);
    // State for this component only, not shared with others
    this.state = {
      isFetching: false, // Control the loading animation
      showNYT: false,
      isOffline: !window.navigator.onLine
    }
  }
 
 
  componentDidMount(){
    window.addEventListener('online', () => {console.log('Became online'); window.location.reload(); });
    window.addEventListener('offline', () => {console.log('Became offline'); this.setState({"isOffline": true});});
    this.loadRecommedations();
  }
 
  loadRecommedations() {
    const device = this.props.device;
    const customer = this.props.customer;
    // if (this.randomBool()) {
    if (true && !(window.APP_VERSION === '4.9.2' && (window.location.hash.match('device-home')) && !this.props.customer.childProfile)) {
      this.loadF1({
        countryCode: customer.country,
        accountId: customer.accountId,
        profileId: customer.profileId,
        productDeviceId: device.productDeviceId
      });
    }
    else {
      this.loadBYR({
        countryCode: customer.country,
        accountId: customer.accountId,
        profileId: customer.profileId,
        productDeviceId: device.productDeviceId,
        size: 4
      });
    }
    if (window.location.href.match(/staging/)) {
      window.Device.showToast({ message: 'Staging Home version: ' + process.env.REACT_APP_VERSION })
    }
  }
 
 
  loadF1(urlParams) {
    
    // Load Recommendations Carousel
    // This data will be cached in case the user views Home offline (native code
    // does caching to allow the page to be viewed when offline).
    this.setState({ isFetching: true });
    fetchF1(urlParams, (err, resp) => {
      
      if (err || !resp) {
        console.error('Error fetching F1 recommendations.', err);
        resp = { eans:[], recommendations:{} };
      }
      
      else {
        this.setState({ isFetching: false });
      }
 
      const data = {
        data: resp.recommendations,
        eanData: resp.eans
      };
      this.props.setF1(data); // Trigger a redux dispatch
      /////////////////////////////////////////////////////////////////
      //
      // New code
      //
      if (err || !resp || !resp.eans || !resp.recommendations) {
        const listParams = Object.assign(urlParams, {
          category_id: '10000000031', // NYT Bestsellers
          offset: 0,
          itemCount: 6 // Might only need 4?
        });
 
        fetchList(listParams, (err, data) => {
          if (err || !data.eans || !data.listData) {
            return; // Ignore and give up (show the same thing we would show if we didn't have this carousel)
          }
          data.listData.title = textReplace(KEY.recommended_for_you);
          const NYTData = {
            eanData: data.eans,
            data: data.listData,
          };
          
          this.setState({ showNYT: true, NYTData });
        });
 
      }
      //
      // New code
      //
      //////////////////////////////////////////////////////////////////////
    }, {
      offlineCache: true,
      size: 4
    });
  }
 
 
  loadBYR(urlParams) {
 
    // Load Recommendations Carousel
    this.setState({ isFetching: true });
    fetchRecommendByRecentEan(urlParams, (err, resp) => {
      if (err) {
        console.error('Error fetching Because You Read recommendations.');
        resp = {eans: []};
      }
      else {
        this.setState({ isFetching: false });
      }
 
      const data = {
        data: resp,
        eanData: {}
      };
      this.props.setBYR(data); // Trigger a redux dispatch
    });
  }
   
 
  render() {
    if(!window.shopPermissionEnable) {
      return <div></div>;
    }
 
    if (this.state.isFetching) {
      return <div>
        <Spinner />
      </div>;
    }
    else if (this.state.isOffline) {
      return <div>
        <EpdHomeOffline />
      </div>
    }
 
    // Only one of these will have data
    let data = this.props.data.f1 !== undefined && this.props.data.f1.eans ? this.props.data.f1 : {};
    data = this.props.data.becauseYouRead && this.props.data.becauseYouRead.eans ? this.props.data.becauseYouRead : data;
 
		// We don't want a scrolling carousel. Limit the number of covers shown
    let carouselData = Object.assign({}, data);
    if (carouselData.eans) {
      carouselData.eans = carouselData.eans.slice(0, 4);
    } 
    console.log('window.APP_VERSION EPD Home111',window.APP_VERSION);
 
    if (this.state.showNYT === true && this.state.NYTData) {
      console.log('showNYT')
      return <div className="epd-home nyt-bestsellsers">
        <ListCarousel 
          loading={ false }
          index={ 1 }
          list={ this.state.NYTData.data }
          eanData={ this.state.NYTData.eanData }
          onCoverClick={ data => this.prepareAnalytics('f1-cover', data) } 
          onListClick={ () => {/*IGNORE*/} } />
      </div>;
    }
    
 
		return (
      <div className="epd-home">
      {window.APP_VERSION === '4.9.2' && (window.location.hash.match('device-home') && !this.props.customer.childProfile)  ? <EpdUpgrade /> :
        <F1Recommendations
          isFetching={ this.state.isFetching }
          data={ carouselData }
          eanData={ this.props.eanData }
          onCoverClick={ data => this.prepareAnalytics('f1-cover', data) } />
      }
      </div>
    );
  }
