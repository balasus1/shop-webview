import React, { Component } from 'react'
 
// Rating Star Images
import fullStar from '../../images/rate_star_on.png'
import halfStar from '../../images/rate_star_half.png'
import noStar from '../../images/rate_star_off.png'
// Rating Star Images EPD
import fullStarEpd from '../../images/rate_star_on_epd.png'
import halfStarEpd from '../../images/rate_star_half_epd.png'
import noStarEpd from '../../images/rate_star_off_epd.png'
 
import '../../style-dist/components/RatingStars.css'
import textReplace from '../lib/i18n'
import KEY from '../lib/constants/languages/keys'
 
 
class RatingCover extends Component {
 
  render() {
    const platform = window.Device.shopPlatform;
    const ratingfullStar = platform === "android"  ||  platform === "ios" ? fullStar : fullStarEpd;
    const ratinghalfStar = platform === "android" ||  platform === "ios" ? halfStar : halfStarEpd;
    const ratingnoStar = platform === "android" ||  platform === "ios"? noStar : noStarEpd;
    const rating = this.props.rating > 0 ? this.props.rating : 0;
 
    let star1 = ratingnoStar;
    let star2 = ratingnoStar;
    let star3 = ratingnoStar;
    let star4 = ratingnoStar;
    let star5 = ratingnoStar;
 
    if (rating >= 4.25) star5 = rating >= 4.75 ? ratingfullStar : ratinghalfStar;
    if (rating >= 3.25) star4 = rating >= 3.75 ? ratingfullStar : ratinghalfStar;
    if (rating >= 2.25) star3 = rating >= 2.75 ? ratingfullStar : ratinghalfStar;
    if (rating >= 1.25) star2 = rating >= 1.75 ? ratingfullStar : ratinghalfStar;
    if (rating >= 0.25) star1 = rating >= 0.75 ? ratingfullStar : ratinghalfStar;
 
    const stars =  [star1, star2, star3, star4, star5].map( (star, i) => {
      return <img key={i} alt="" src={ star } />;
    });
 
    return (
      <div className="rating-stars">
        <span role="img"
              className="rating-star" 
              title={ `${textReplace(KEY.rating)}: ${rating} ${textReplace(KEY.of)} 5` }
              tabIndex="0">{ stars }</span>
      </div>
    );
  }
 
 
}
 
export default RatingCover;
