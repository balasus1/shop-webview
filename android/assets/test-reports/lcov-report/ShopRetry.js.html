import React, { Component } from 'react'
 
import '../../style-dist/components/ShopRetry.css'
import {constants} from "../lib/utils";
import textReplace from '../lib/i18n';
import KEY from '../lib/constants/languages/keys';
class ShopRetry extends Component {
  retry() {
    window.Device.reload();
  }
 
  render() {
    const message = (this.props.status === true) ?
        constants.APPLICATION_MESSAGES.NO_DATA :
        constants.APPLICATION_MESSAGES.ERROR;
    return (
            <div className='retry-container'>
                <div className="shop-reload">
                  <p>{ message }</p>
                  <div className="retry-cta" onClick={this.retry}>
                    {textReplace(KEY.retry)}
                  </div>
                </div>
            </div>
    );
  }
}
 
export default ShopRetry;
